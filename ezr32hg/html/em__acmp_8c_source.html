114 void ACMP_CapsenseInit(ACMP_TypeDef *acmp, const ACMP_CapsenseInit_TypeDef *init)
115 {
116  /* Make sure the module exists on the selected chip */
117  EFM_ASSERT(ACMP_REF_VALID(acmp));
118 
119  /* Make sure that vddLevel is within bounds */
120 #if defined(_ACMP_INPUTSEL_VDDLEVEL_MASK)
121  EFM_ASSERT(init->vddLevel < 64);
122 #else
123  EFM_ASSERT(init->vddLevelLow < 64);
124  EFM_ASSERT(init->vddLevelHigh < 64);
125 #endif
126 
127  /* Make sure biasprog is within bounds */
128  EFM_ASSERT(init->biasProg <=
129  (_ACMP_CTRL_BIASPROG_MASK >> _ACMP_CTRL_BIASPROG_SHIFT));
130 
131  /* Set control register. No need to set interrupt modes */
132  acmp->CTRL = (init->fullBias << _ACMP_CTRL_FULLBIAS_SHIFT)
133 #if defined(_ACMP_CTRL_HALFBIAS_MASK)
134  | (init->halfBias << _ACMP_CTRL_HALFBIAS_SHIFT)
135 #endif
136  | (init->biasProg << _ACMP_CTRL_BIASPROG_SHIFT)
137 #if defined(_ACMP_CTRL_WARMTIME_MASK)
138  | (init->warmTime << _ACMP_CTRL_WARMTIME_SHIFT)
139 #endif
140 #if defined(_ACMP_CTRL_HYSTSEL_MASK)
141  | (init->hysteresisLevel << _ACMP_CTRL_HYSTSEL_SHIFT)
142 #endif
143 #if defined(_ACMP_CTRL_ACCURACY_MASK)
144  | ACMP_CTRL_ACCURACY_HIGH
145 #endif
146  ;
147 
148 #if defined(_ACMP_HYSTERESIS0_MASK)
149  acmp->HYSTERESIS0 = (init->vddLevelHigh << _ACMP_HYSTERESIS0_DIVVA_SHIFT)
150  | (init->hysteresisLevel_0 << _ACMP_HYSTERESIS0_HYST_SHIFT);
151  acmp->HYSTERESIS1 = (init->vddLevelLow << _ACMP_HYSTERESIS1_DIVVA_SHIFT)
152  | (init->hysteresisLevel_1 << _ACMP_HYSTERESIS1_HYST_SHIFT);
153 #endif
154

156  acmp->INPUTSEL = (init->resistor << _ACMP_INPUTSEL_CSRESSEL_SHIFT)
157  | ACMP_INPUTSEL_CSRESEN
158 #if defined(_ACMP_INPUTSEL_LPREF_MASK)
159  | (init->lowPowerReferenceEnabled << _ACMP_INPUTSEL_LPREF_SHIFT)
160 #endif
161 #if defined(_ACMP_INPUTSEL_VDDLEVEL_MASK)

163 #endif
164 #if defined(ACMP_INPUTSEL_NEGSEL_CAPSENSE)
165  | ACMP_INPUTSEL_NEGSEL_CAPSENSE
166 #else
167  | ACMP_INPUTSEL_VASEL_VDD
168  | ACMP_INPUTSEL_NEGSEL_VADIV
169 #endif
170  ;
171 
172  /* Enable ACMP if requested. */
173  BUS_RegBitWrite(&(acmp->CTRL), _ACMP_CTRL_EN_SHIFT, init->enable);
