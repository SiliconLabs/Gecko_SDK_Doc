192 static void emuRestore(void)
193 {
194  uint32_t oscEnCmd;
195  uint32_t cmuLocked;
196 
197  /* Although we could use the CMU API for most of the below handling, we */
198  /* would like this function to be as efficient as possible. */
199 
200  /* CMU registers may be locked */
201  cmuLocked = CMU->LOCK & CMU_LOCK_LOCKKEY_LOCKED;
202  CMU_Unlock();
203 
204  /* AUXHFRCO are automatically disabled (except if using debugger). */
205  /* HFRCO, USHFRCO and HFXO are automatically disabled. */
206  /* LFRCO/LFXO may be disabled by SW in EM3. */
207  /* Restore according to status prior to entering energy mode. */
208  oscEnCmd = 0;
209  oscEnCmd |= ((cmuStatus & CMU_STATUS_HFRCOENS) ? CMU_OSCENCMD_HFRCOEN : 0);
210  oscEnCmd |= ((cmuStatus & CMU_STATUS_AUXHFRCOENS) ? CMU_OSCENCMD_AUXHFRCOEN : 0);
211  oscEnCmd |= ((cmuStatus & CMU_STATUS_LFRCOENS) ? CMU_OSCENCMD_LFRCOEN : 0);
212  oscEnCmd |= ((cmuStatus & CMU_STATUS_HFXOENS) ? CMU_OSCENCMD_HFXOEN : 0);
213  oscEnCmd |= ((cmuStatus & CMU_STATUS_LFXOENS) ? CMU_OSCENCMD_LFXOEN : 0);
214 #if defined( _CMU_STATUS_USHFRCOENS_MASK )
215  oscEnCmd |= ((cmuStatus & CMU_STATUS_USHFRCOENS) ? CMU_OSCENCMD_USHFRCOEN : 0);
216 #endif
217  CMU->OSCENCMD = oscEnCmd;
218 
219 
220 #if defined( _CMU_HFCLKSTATUS_RESETVALUE )
221  /* Restore oscillator used for clocking core */
222  switch (cmuHfclkStatus & _CMU_HFCLKSTATUS_SELECTED_MASK)
223  {
224  case CMU_HFCLKSTATUS_SELECTED_LFRCO:
225  /* HFRCO could only be selected if the autostart HFXO feature is not
226  * enabled, otherwise the HFXO would be started and selected automatically.
227  * Note: this error hook helps catching erroneous oscillator configurations,
228  * when the AUTOSTARTSELEM0EM1 is set in CMU_HFXOCTRL. */
229  if (!(CMU->HFXOCTRL & CMU_HFXOCTRL_AUTOSTARTSELEM0EM1))
230  {
231  /* Wait for LFRCO to stabilize */
232  while (!(CMU->STATUS & CMU_STATUS_LFRCORDY))
233  ;
234  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_LFRCO;
235  }
236  else
237  {
238  EFM_ASSERT(0);
239  }
240  break;
241 
242  case CMU_HFCLKSTATUS_SELECTED_LFXO:
243  /* Wait for LFXO to stabilize */
244  while (!(CMU->STATUS & CMU_STATUS_LFXORDY))
245  ;
246  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_LFXO;
247  break;
248 
249  case CMU_HFCLKSTATUS_SELECTED_HFXO:
250  /* Wait for HFXO to stabilize */
251  while (!(CMU->STATUS & CMU_STATUS_HFXORDY))
252  ;
253 #if defined( _CMU_HFXOCTRL_PEAKDETSHUNTOPTMODE_MASK )
254  if (BUS_RegMaskedRead(&CMU->HFXOCTRL,
255  _CMU_HFXOCTRL_PEAKDETSHUNTOPTMODE_MASK)
256  == CMU_HFXOCTRL_PEAKDETSHUNTOPTMODE_AUTOCMD)
257  {
258  while (BUS_RegMaskedRead(&CMU->STATUS,
259  _CMU_STATUS_HFXOSHUNTOPTRDY_MASK
260  | _CMU_STATUS_HFXOPEAKDETRDY_MASK)
261  != (CMU_STATUS_HFXOSHUNTOPTRDY | CMU_STATUS_HFXOPEAKDETRDY))
262  ;
263  }
264 #endif
265  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_HFXO;
266  break;
267 
268  default: /* CMU_HFCLKSTATUS_SELECTED_HFRCO */
269  /* If core clock was HFRCO core clock, it is automatically restored to */
270  /* state prior to entering energy mode. No need for further action. */
271  break;
272  }
273 #else
274  switch (cmuStatus & (CMU_STATUS_HFRCOSEL
275  | CMU_STATUS_HFXOSEL
276  | CMU_STATUS_LFRCOSEL
277 #if defined( CMU_STATUS_USHFRCODIV2SEL )
278  | CMU_STATUS_USHFRCODIV2SEL
279 #endif
280  | CMU_STATUS_LFXOSEL))
281  {
282  case CMU_STATUS_LFRCOSEL:
283  /* Wait for LFRCO to stabilize */
284  while (!(CMU->STATUS & CMU_STATUS_LFRCORDY))
285  ;
286  CMU->CMD = CMU_CMD_HFCLKSEL_LFRCO;
287  break;
288 
289  case CMU_STATUS_LFXOSEL:
290  /* Wait for LFXO to stabilize */
291  while (!(CMU->STATUS & CMU_STATUS_LFXORDY))
292  ;
293  CMU->CMD = CMU_CMD_HFCLKSEL_LFXO;
294  break;
295 
296  case CMU_STATUS_HFXOSEL:
297  /* Wait for HFXO to stabilize */
298  while (!(CMU->STATUS & CMU_STATUS_HFXORDY))
299  ;
300  CMU->CMD = CMU_CMD_HFCLKSEL_HFXO;
301  break;
302 
303 #if defined( CMU_STATUS_USHFRCODIV2SEL )
304  case CMU_STATUS_USHFRCODIV2SEL:
305  /* Wait for USHFRCO to stabilize */
306  while (!(CMU->STATUS & CMU_STATUS_USHFRCORDY))
307  ;
308  CMU->CMD = _CMU_CMD_HFCLKSEL_USHFRCODIV2;
309  break;
310 #endif
311 
312  default: /* CMU_STATUS_HFRCOSEL */
313  /* If core clock was HFRCO core clock, it is automatically restored to */
314  /* state prior to entering energy mode. No need for further action. */
315  break;
316  }
317 
318  /* If HFRCO was disabled before entering Energy Mode, turn it off again */
319  /* as it is automatically enabled by wake up */
320  if ( ! (cmuStatus & CMU_STATUS_HFRCOENS) )
321  {
322  CMU->OSCENCMD = CMU_OSCENCMD_HFRCODIS;
323  }
324 #endif
325  /* Restore CMU register locking */
326  if (cmuLocked)
327  {
328  CMU_Lock();
329  }
330 }
