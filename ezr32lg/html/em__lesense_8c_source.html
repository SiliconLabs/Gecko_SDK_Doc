212 uint32_t LESENSE_ScanFreqSet(uint32_t refFreq, uint32_t const scanFreq)
213 {
214  uint32_t tmp;
215  uint32_t pcPresc = 0UL; /* Period counter prescaler. */
216  uint32_t clkDiv = 1UL; /* Clock divisor value (2^pcPresc). */
217  uint32_t pcTop = 63UL; /* Period counter top value (max. 63). */
218  uint32_t calcScanFreq; /* Variable for testing the calculation algorithm. */
219 
220 
221  /* If refFreq is set to 0, the currently configured reference clock is
222  * assumed. */
223  if (!refFreq)
224  {
225  refFreq = CMU_ClockFreqGet(cmuClock_LESENSE);
226  }
227 
228  /* Max. value of pcPresc is 128, thus using reference frequency less than
229  * 33554431Hz (33.554431MHz), the frequency calculation in the while loop
230  * below will not overflow. */
231  EFM_ASSERT(refFreq < ((uint32_t)UINT32_MAX / 128UL));
232 
233  /* Sanity check of scan frequency value. */
234  EFM_ASSERT((scanFreq > 0U) && (scanFreq <= refFreq));
235 
236  /* Calculate the minimum necessary prescaler value in order to provide the
237  * biggest possible resolution for setting scan frequency.
238  * Maximum number of calculation cycles is 7 (value of lesenseClkDiv_128). */
239  while ((refFreq / ((uint32_t)scanFreq * clkDiv) > (pcTop + 1UL))
240  && (pcPresc < lesenseClkDiv_128))
241  {
242  ++pcPresc;
243  clkDiv = (uint32_t)1UL << pcPresc;
244  }
245 
246  /* Calculate pcTop value. */
247  pcTop = ((uint32_t)refFreq / ((uint32_t)scanFreq * clkDiv)) - 1UL;
248 
249  /* Clear current PCPRESC and PCTOP settings. Be aware of the effect of
250  * non-atomic Read-Modify-Write on LESENSE->TIMCRTL. */
251  tmp = LESENSE->TIMCTRL & (~_LESENSE_TIMCTRL_PCPRESC_MASK
252  & ~_LESENSE_TIMCTRL_PCTOP_MASK);
253 
254  /* Set new values in tmp while reserving other settings. */
255  tmp |= ((uint32_t)pcPresc << _LESENSE_TIMCTRL_PCPRESC_SHIFT)
256  | ((uint32_t)pcTop << _LESENSE_TIMCTRL_PCTOP_SHIFT);
257 
258  /* Set values in LESENSE_TIMCTRL register. */
259  LESENSE->TIMCTRL = tmp;
260 
261  /* For testing the calculation algorithm. */
262  calcScanFreq = ((uint32_t)refFreq / ((uint32_t)(1UL + pcTop) * clkDiv));
263 
264  return calcScanFreq;
265 }

575 void LESENSE_AltExConfig(LESENSE_ConfAltEx_TypeDef const *confAltEx)
576 {
577  uint32_t i;
578  uint32_t tmp;
579 
580 
581  /* Configure alternate excitation mapping.
582  * Atomic read-modify-write using BUS_RegBitWrite function in order to
583  * support reconfiguration during LESENSE operation. */
584  BUS_RegBitWrite(&(LESENSE->CTRL),
585  _LESENSE_CTRL_ALTEXMAP_SHIFT,
586  confAltEx->altExMap);
587 
588  switch (confAltEx->altExMap)
589  {
590  case lesenseAltExMapALTEX:
591  /* Iterate through the 8 possible alternate excitation pin descriptors. */
592  for (i = 0U; i < 8U; ++i)
593  {
594  /* Enable/disable alternate excitation pin i.
595  * Atomic read-modify-write using BUS_RegBitWrite function in order to
596  * support reconfiguration during LESENSE operation. */
597  BUS_RegBitWrite(&(LESENSE->ROUTE),
598  (16UL + i),
599  confAltEx->AltEx[i].enablePin);
600 
601  /* Setup the idle phase state of alternate excitation pin i.
602  * Read-modify-write in order to support reconfiguration during LESENSE
603  * operation. */
604  tmp = (LESENSE->ALTEXCONF & ~((uint32_t)0x3UL << (i * 2UL)));
605  tmp |= ((uint32_t)confAltEx->AltEx[i].idleConf << (i * 2UL));
606  LESENSE->ALTEXCONF = tmp;
607 
608  /* Enable/disable always excite on channel i */
609  BUS_RegBitWrite(&(LESENSE->ALTEXCONF),
610  (16UL + i),
611  confAltEx->AltEx[i].alwaysEx);
612  }
613  break;
614 
615  case lesenseAltExMapACMP:
616  /* Iterate through all the 16 alternate excitation channels */
617  for (i = 0U; i < 16U; ++i)
618  {
619  /* Enable/disable alternate ACMP excitation channel pin i. */
620  /* Atomic read-modify-write using BUS_RegBitWrite function in order to
621  * support reconfiguration during LESENSE operation. */
622  BUS_RegBitWrite(&(LESENSE->ROUTE),
623  i,
624  confAltEx->AltEx[i].enablePin);
625  }
626  break;
627  default:
628  /* Illegal value. */
629  EFM_ASSERT(0);
630  break;
631  }
