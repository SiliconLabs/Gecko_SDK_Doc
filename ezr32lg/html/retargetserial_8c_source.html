210 int RETARGET_ReadChar(void)
211 {
212  int c = -1;
213 
214  if (initialized == false)
215  {
216  RETARGET_SerialInit();
217  }
218 
219  INT_Disable();
220  if (rxCount > 0)

227  }
228  rxCount--;
229  }
230  INT_Enable();
