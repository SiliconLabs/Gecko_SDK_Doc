218 void RMU_ResetControl(RMU_Reset_TypeDef reset, RMU_ResetMode_TypeDef mode)
219 {
220  /* Note that the RMU supports bit-band access, but not peripheral bit-field set/clear */
221 #if defined(_RMU_CTRL_PINRMODE_MASK)
222  uint32_t val;
223 #endif
224  uint32_t shift;
225 
226  shift = SL_CTZ((uint32_t)reset);
227 #if defined(_RMU_CTRL_PINRMODE_MASK)
228  val = (uint32_t)mode << shift;
229  RMU->CTRL = (RMU->CTRL & ~reset) | val;
230 #else
231  BUS_RegBitWrite(&RMU->CTRL, (uint32_t)shift, mode ? 1 : 0);
232 #endif
233 }

287 uint32_t RMU_ResetCauseGet(void)
288 {
289 #define LB_CLW0 (* (volatile uint32_t *)(LOCKBITS_BASE + 122))
290 #define LB_CLW0_PINRESETSOFT (1 << 2)
291 
292 
293 #if !defined(EMLIB_REGRESSION_TEST)
294  uint32_t rstCause = RMU->RSTCAUSE;
295 #endif
296  uint32_t validRstCause = 0;
297  uint32_t zeroXMask;
298  uint32_t i;
299 
300  for (i = 0; i < NUM_RSTCAUSES; i++)
301  {
302  zeroXMask = resetCauseMasks[i].resetCauseZeroXMask;
303 #if defined( _SILICON_LABS_32B_PLATFORM_2 )
304  /* Handle soft/hard pin reset */
305  if (!(LB_CLW0 & LB_CLW0_PINRESETSOFT))
306  {
307  /* RSTCAUSE_EXTRST must be 0 if pin reset is configured as hard reset */
308  switch (resetCauseMasks[i].resetCauseMask)
309  {
310  case RMU_RSTCAUSE_LOCKUPRST:
311  /* Fallthrough */
312  case RMU_RSTCAUSE_SYSREQRST:
313  /* Fallthrough */
314  case RMU_RSTCAUSE_WDOGRST:
315  /* Fallthrough */
316  case RMU_RSTCAUSE_EM4RST:
317  zeroXMask |= RMU_RSTCAUSE_EXTRST;
318  break;
319  }
320  }
321 #endif
322 
323  /* Check reset cause requirements. Note that a bit is "don't care" if 0 in
324  both resetCauseMask and resetCauseZeroXMask. */
325  if ((rstCause & resetCauseMasks[i].resetCauseMask)
326  && !(rstCause & zeroXMask))
327  {
328  /* Add this reset-cause to the mask of qualified reset-causes */
329  validRstCause |= resetCauseMasks[i].resetCauseMask;
330  }
331  }
332  return validRstCause;
333 }
