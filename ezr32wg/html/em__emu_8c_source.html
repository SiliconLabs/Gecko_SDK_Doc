167 static void emuRestore(void)
168 {
169  uint32_t oscEnCmd;
170  uint32_t cmuLocked;
171 
172  /* Although we could use the CMU API for most of the below handling, we */
173  /* would like this function to be as efficient as possible. */
174 
175  /* CMU registers may be locked */
176  cmuLocked = CMU->LOCK & CMU_LOCK_LOCKKEY_LOCKED;
177  CMU_Unlock();
178 
179  /* AUXHFRCO are automatically disabled (except if using debugger). */
180  /* HFRCO, USHFRCO and HFXO are automatically disabled. */
181  /* LFRCO/LFXO may be disabled by SW in EM3. */
182  /* Restore according to status prior to entering energy mode. */
183  oscEnCmd = 0;
184  oscEnCmd |= ((cmuStatus & CMU_STATUS_HFRCOENS) ? CMU_OSCENCMD_HFRCOEN : 0);
185  oscEnCmd |= ((cmuStatus & CMU_STATUS_AUXHFRCOENS) ? CMU_OSCENCMD_AUXHFRCOEN : 0);
186  oscEnCmd |= ((cmuStatus & CMU_STATUS_LFRCOENS) ? CMU_OSCENCMD_LFRCOEN : 0);
187  oscEnCmd |= ((cmuStatus & CMU_STATUS_HFXOENS) ? CMU_OSCENCMD_HFXOEN : 0);
188  oscEnCmd |= ((cmuStatus & CMU_STATUS_LFXOENS) ? CMU_OSCENCMD_LFXOEN : 0);
189 #if defined( _CMU_STATUS_USHFRCOENS_MASK )
190  oscEnCmd |= ((cmuStatus & CMU_STATUS_USHFRCOENS) ? CMU_OSCENCMD_USHFRCOEN : 0);
191 #endif
192  CMU->OSCENCMD = oscEnCmd;
193 
194 
195 #if defined( _CMU_HFCLKSTATUS_RESETVALUE )
196  /* Restore oscillator used for clocking core */
197  switch (cmuHfclkStatus & _CMU_HFCLKSTATUS_SELECTED_MASK)
198  {
199  case CMU_HFCLKSTATUS_SELECTED_LFRCO:
200  /* HFRCO could only be selected if the autostart HFXO feature is not
201  * enabled, otherwise the HFXO would be started and selected automatically.
202  * Note: this error hook helps catching erroneous oscillator configurations,
203  * when the AUTOSTARTSELEM0EM1 is set in CMU_HFXOCTRL. */
204  if (!(CMU->HFXOCTRL & CMU_HFXOCTRL_AUTOSTARTSELEM0EM1))
205  {
206  /* Wait for LFRCO to stabilize */
207  while (!(CMU->STATUS & CMU_STATUS_LFRCORDY))
208  ;
209  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_LFRCO;
210  }
211  else
212  {
213  EFM_ASSERT(0);
214  }
215  break;
216 
217  case CMU_HFCLKSTATUS_SELECTED_LFXO:
218  /* Wait for LFXO to stabilize */
219  while (!(CMU->STATUS & CMU_STATUS_LFXORDY))
220  ;
221  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_LFXO;
222  break;
223 
224  case CMU_HFCLKSTATUS_SELECTED_HFXO:
225  /* Wait for HFXO to stabilize */
226  while (!(CMU->STATUS & CMU_STATUS_HFXORDY))
227  ;
228 #if defined( _CMU_HFXOCTRL_PEAKDETSHUNTOPTMODE_MASK )
229  if (BUS_RegMaskedRead(&CMU->HFXOCTRL,
230  _CMU_HFXOCTRL_PEAKDETSHUNTOPTMODE_MASK)
231  == CMU_HFXOCTRL_PEAKDETSHUNTOPTMODE_AUTOCMD)
232  {
233  while (BUS_RegMaskedRead(&CMU->STATUS,
234  _CMU_STATUS_HFXOSHUNTOPTRDY_MASK
235  | _CMU_STATUS_HFXOPEAKDETRDY_MASK)
236  != (CMU_STATUS_HFXOSHUNTOPTRDY | CMU_STATUS_HFXOPEAKDETRDY))
237  ;
238  }
239 #endif
240  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_HFXO;
241  break;
242 
243  default: /* CMU_HFCLKSTATUS_SELECTED_HFRCO */
244  /* If core clock was HFRCO core clock, it is automatically restored to */
245  /* state prior to entering energy mode. No need for further action. */
246  break;
247  }
248 #else
249  switch (cmuStatus & (CMU_STATUS_HFRCOSEL
250  | CMU_STATUS_HFXOSEL
251  | CMU_STATUS_LFRCOSEL
252 #if defined( CMU_STATUS_USHFRCODIV2SEL )
253  | CMU_STATUS_USHFRCODIV2SEL
254 #endif
255  | CMU_STATUS_LFXOSEL))
256  {
257  case CMU_STATUS_LFRCOSEL:
258  /* Wait for LFRCO to stabilize */
259  while (!(CMU->STATUS & CMU_STATUS_LFRCORDY))
260  ;
261  CMU->CMD = CMU_CMD_HFCLKSEL_LFRCO;
262  break;
263 
264  case CMU_STATUS_LFXOSEL:
265  /* Wait for LFXO to stabilize */
266  while (!(CMU->STATUS & CMU_STATUS_LFXORDY))
267  ;
268  CMU->CMD = CMU_CMD_HFCLKSEL_LFXO;
269  break;
270 
271  case CMU_STATUS_HFXOSEL:
272  /* Wait for HFXO to stabilize */
273  while (!(CMU->STATUS & CMU_STATUS_HFXORDY))
274  ;
275  CMU->CMD = CMU_CMD_HFCLKSEL_HFXO;
276  break;
277 
278 #if defined( CMU_STATUS_USHFRCODIV2SEL )
279  case CMU_STATUS_USHFRCODIV2SEL:
280  /* Wait for USHFRCO to stabilize */
281  while (!(CMU->STATUS & CMU_STATUS_USHFRCORDY))
282  ;
283  CMU->CMD = _CMU_CMD_HFCLKSEL_USHFRCODIV2;
284  break;
285 #endif
286 
287  default: /* CMU_STATUS_HFRCOSEL */
288  /* If core clock was HFRCO core clock, it is automatically restored to */
289  /* state prior to entering energy mode. No need for further action. */
290  break;
291  }
292 
293  /* If HFRCO was disabled before entering Energy Mode, turn it off again */
294  /* as it is automatically enabled by wake up */
295  if ( ! (cmuStatus & CMU_STATUS_HFRCOENS) )
296  {
297  CMU->OSCENCMD = CMU_OSCENCMD_HFRCODIS;
298  }
299 #endif
300  /* Restore CMU register locking */
301  if (cmuLocked)
302  {
303  CMU_Lock();
304  }
305 }

654 void EMU_EnterEM4(void)
655 {
656  int i;
657 
658 #if defined( _EMU_EM4CTRL_EM4ENTRY_SHIFT )
659  uint32_t em4seq2 = (EMU->EM4CTRL & ~_EMU_EM4CTRL_EM4ENTRY_MASK)
660  | (2 << _EMU_EM4CTRL_EM4ENTRY_SHIFT);
661  uint32_t em4seq3 = (EMU->EM4CTRL & ~_EMU_EM4CTRL_EM4ENTRY_MASK)
662  | (3 << _EMU_EM4CTRL_EM4ENTRY_SHIFT);
663 #else
664  uint32_t em4seq2 = (EMU->CTRL & ~_EMU_CTRL_EM4CTRL_MASK)
665  | (2 << _EMU_CTRL_EM4CTRL_SHIFT);
666  uint32_t em4seq3 = (EMU->CTRL & ~_EMU_CTRL_EM4CTRL_MASK)
667  | (3 << _EMU_CTRL_EM4CTRL_SHIFT);
668 #endif
669 
670  /* Make sure register write lock is disabled */
671  EMU_Unlock();
672 
673 #if defined( _EMU_EM4CTRL_MASK )
674  if ((EMU->EM4CTRL & _EMU_EM4CTRL_EM4STATE_MASK) == EMU_EM4CTRL_EM4STATE_EM4S)
675  {
676  uint32_t dcdcMode = EMU->DCDCCTRL & _EMU_DCDCCTRL_DCDCMODE_MASK;
677  if (dcdcMode == EMU_DCDCCTRL_DCDCMODE_LOWNOISE
678  || dcdcMode == EMU_DCDCCTRL_DCDCMODE_LOWPOWER)
679  {
680  /* DCDC is not supported in EM4S so we switch DCDC to bypass mode before
681  * entering EM4S */
682  EMU_DCDCModeSet(emuDcdcMode_Bypass);
683  }
684  }
685 #endif
686 
687 #if defined( _EMU_EM4CTRL_MASK ) && defined( ERRATA_FIX_EMU_E208_EN )
688  if (EMU->EM4CTRL & EMU_EM4CTRL_EM4STATE_EM4H)
689  {
690  /* Fix for errata EMU_E208 - Occasional Full Reset After Exiting EM4H.
691  * Full description of errata fix can be found in the errata document. */
692  *(volatile uint32_t *)(EMU_BASE + 0x190) = 0x0000ADE8UL;
693  *(volatile uint32_t *)(EMU_BASE + 0x198) |= (0x1UL << 7);
694  }
695 #endif
696 
697 #if defined( ERRATA_FIX_EMU_E108_EN )
698  /* Fix for errata EMU_E108 - High Current Consumption on EM4 Entry. */
699  __disable_irq();
700  *(volatile uint32_t *)0x400C80E4 = 0;
701 #endif
702 
703 #if defined( _EMU_DCDCCTRL_MASK )
704  dcdcFetCntSet(true);
705  dcdcHsFixLnBlock();
706 #endif
707 
708  for (i = 0; i < 4; i++)
709  {
710 #if defined( _EMU_EM4CTRL_EM4ENTRY_SHIFT )
711  EMU->EM4CTRL = em4seq2;
712  EMU->EM4CTRL = em4seq3;
713  }
714  EMU->EM4CTRL = em4seq2;
715 #else
716  EMU->CTRL = em4seq2;
717  EMU->CTRL = em4seq3;
718  }
719  EMU->CTRL = em4seq2;
720 #endif
721 }

1143  uint32_t zdetLimSel;
1144  uint32_t pFetCnt;
1145  uint16_t maxReverseCurrent_mA;

1147  /* 80mA as recommended peak in Application Note AN0948.
1148  The peak current is the average current plus 50% of the current ripple.
1149  Hence, a 14mA average current is recommended in LP mode. Since LP PFETCNT is also
1150  a constant, we get lpcLimImSel = 1. The following calculation is provided
1151  for documentation only. */
1152  const uint32_t lpcLim = (((14 + 40) + ((14 + 40) / 2))
1153  / (5 * (DCDC_LP_PFET_CNT + 1)))
1154  - 1;

1159  >> _EMU_DCDCMISCCTRL_PFETCNT_SHIFT;
1160 
1161  /* Compute LN current limiter threshold from nominal user input current and
1162  LN PFETCNT as described in the register description for
1163  EMU_DCDCMISCCTRL_LNCLIMILIMSEL. */
1164  lncLimSel = (((dcdcMaxCurrent_mA + 40) + ((dcdcMaxCurrent_mA + 40) / 2))
1165  / (5 * (pFetCnt + 1)))
1166  - 1;
1167

1169  lncLimSel = SL_MIN(lncLimSel,
1170  _EMU_DCDCMISCCTRL_LNCLIMILIMSEL_MASK
1171  >> _EMU_DCDCMISCCTRL_LNCLIMILIMSEL_SHIFT);
1172

1174

1176  EFM_ASSERT((lncLimSel & ~_EMU_DCDCMISCCTRL_LNCLIMILIMSEL_MASK) == 0x0);
1177  EFM_ASSERT((lpcLimSel & ~_EMU_DCDCMISCCTRL_LPCLIMILIMSEL_MASK) == 0x0);
