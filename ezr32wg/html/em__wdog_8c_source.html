106 void WDOGn_Feed(WDOG_TypeDef *wdog)
107 {
108  /* The watchdog should not be fed while it is disabled */
109  if (!(wdog->CTRL & WDOG_CTRL_EN))
110  {
111  return;
112  }
113 
114  /* If a previous clearing is being synchronized to LF domain, then there */
115  /* is no point in waiting for it to complete before clearing over again. */
116  /* This avoids stalling the core in the typical use case where some idle loop */
117  /* keeps clearing the watchdog. */
118  if (wdog->SYNCBUSY & WDOG_SYNCBUSY_CMD)
119  {
120  return;
121  }
122  /* Before writing to the WDOG_CMD register we also need to make sure that
123  * any previous write to WDOG_CTRL is complete. */
124  while ( wdog->SYNCBUSY & WDOG_SYNCBUSY_CTRL )
125  ;
126 
127  wdog->CMD = WDOG_CMD_CLEAR;
128 }
