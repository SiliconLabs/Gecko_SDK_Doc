144 static void emuRestore(void)
145 {
146  uint32_t oscEnCmd;
147  uint32_t cmuLocked;
148 
149  /* Although we could use the CMU API for most of the below handling, we */
150  /* would like this function to be as efficient as possible. */
151 
152  /* CMU registers may be locked */
153  cmuLocked = CMU->LOCK & CMU_LOCK_LOCKKEY_LOCKED;
154  CMU_Unlock();
155 
156  /* AUXHFRCO are automatically disabled (except if using debugger). */
157  /* HFRCO, USHFRCO and HFXO are automatically disabled. */
158  /* LFRCO/LFXO may be disabled by SW in EM3. */
159  /* Restore according to status prior to entering energy mode. */
160  oscEnCmd = 0;
161  oscEnCmd |= ((cmuStatus & CMU_STATUS_HFRCOENS) ? CMU_OSCENCMD_HFRCOEN : 0);
162  oscEnCmd |= ((cmuStatus & CMU_STATUS_AUXHFRCOENS) ? CMU_OSCENCMD_AUXHFRCOEN : 0);
163  oscEnCmd |= ((cmuStatus & CMU_STATUS_LFRCOENS) ? CMU_OSCENCMD_LFRCOEN : 0);
164  oscEnCmd |= ((cmuStatus & CMU_STATUS_HFXOENS) ? CMU_OSCENCMD_HFXOEN : 0);
165  oscEnCmd |= ((cmuStatus & CMU_STATUS_LFXOENS) ? CMU_OSCENCMD_LFXOEN : 0);
166 #if defined( _CMU_STATUS_USHFRCOENS_MASK )
167  oscEnCmd |= ((cmuStatus & CMU_STATUS_USHFRCOENS) ? CMU_OSCENCMD_USHFRCOEN : 0);
168 #endif
169  CMU->OSCENCMD = oscEnCmd;
170 
171 
172  /* Restore oscillator used for clocking core */
173 #if defined( _CMU_HFCLKSTATUS_RESETVALUE )
174  switch (cmuHfclkStatus & _CMU_HFCLKSTATUS_SELECTED_MASK)
175  {
176  case CMU_HFCLKSTATUS_SELECTED_LFRCO:
177  /* HFRCO could only be selected if the autostart HFXO feature is not
178  * enabled, otherwise the HFXO would be started and selected automatically.
179  * Note: this error hook helps catching erroneous oscillator configurations,
180  * when the AUTOSTARTSELEM0EM1 is set in CMU_HFXOCTRL. */
181  if (!(CMU->HFXOCTRL & CMU_HFXOCTRL_AUTOSTARTSELEM0EM1))
182  {
183  /* Wait for LFRCO to stabilize */
184  while (!(CMU->STATUS & CMU_STATUS_LFRCORDY))
185  ;
186  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_LFRCO;
187  }
188  else
189  {
190  EFM_ASSERT(0);
191  }
192  break;
193 
194  case CMU_HFCLKSTATUS_SELECTED_LFXO:
195  /* Wait for LFXO to stabilize */
196  while (!(CMU->STATUS & CMU_STATUS_LFXORDY))
197  ;
198  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_LFXO;
199  break;
200 
201  case CMU_HFCLKSTATUS_SELECTED_HFXO:
202  /* Wait for HFXO to stabilize */
203  while (!(CMU->STATUS & CMU_STATUS_HFXORDY))
204  ;
205  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_HFXO;
206  break;
207 
208  default: /* CMU_HFCLKSTATUS_SELECTED_HFRCO */
209  /* If core clock was HFRCO core clock, it is automatically restored to */
210  /* state prior to entering energy mode. No need for further action. */
211  break;
212  }
213 #else
214  switch (cmuStatus & (CMU_STATUS_HFRCOSEL
215  | CMU_STATUS_HFXOSEL
216  | CMU_STATUS_LFRCOSEL
217 #if defined( CMU_STATUS_USHFRCODIV2SEL )
218  | CMU_STATUS_USHFRCODIV2SEL
219 #endif
220  | CMU_STATUS_LFXOSEL))
221  {
222  case CMU_STATUS_LFRCOSEL:
223  /* Wait for LFRCO to stabilize */
224  while (!(CMU->STATUS & CMU_STATUS_LFRCORDY))
225  ;
226  CMU->CMD = CMU_CMD_HFCLKSEL_LFRCO;
227  break;
228 
229  case CMU_STATUS_LFXOSEL:
230  /* Wait for LFXO to stabilize */
231  while (!(CMU->STATUS & CMU_STATUS_LFXORDY))
232  ;
233  CMU->CMD = CMU_CMD_HFCLKSEL_LFXO;
234  break;
235 
236  case CMU_STATUS_HFXOSEL:
237  /* Wait for HFXO to stabilize */
238  while (!(CMU->STATUS & CMU_STATUS_HFXORDY))
239  ;
240  CMU->CMD = CMU_CMD_HFCLKSEL_HFXO;
241  break;
242 
243 #if defined( CMU_STATUS_USHFRCODIV2SEL )
244  case CMU_STATUS_USHFRCODIV2SEL:
245  /* Wait for USHFRCO to stabilize */
246  while (!(CMU->STATUS & CMU_STATUS_USHFRCORDY))
247  ;
248  CMU->CMD = _CMU_CMD_HFCLKSEL_USHFRCODIV2;
249  break;
250 #endif
251 
252  default: /* CMU_STATUS_HFRCOSEL */
253  /* If core clock was HFRCO core clock, it is automatically restored to */
254  /* state prior to entering energy mode. No need for further action. */
255  break;
256  }
257 
258  /* If HFRCO was disabled before entering Energy Mode, turn it off again */
259  /* as it is automatically enabled by wake up */
260  if ( ! (cmuStatus & CMU_STATUS_HFRCOENS) )
261  {
262  CMU->OSCENCMD = CMU_OSCENCMD_HFRCODIS;
263  }
264 #endif
265  /* Restore CMU register locking */
266  if (cmuLocked)
267  {
268  CMU_Lock();
269  }
270 }
