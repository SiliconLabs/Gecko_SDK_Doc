322 static bool ReceiveDmaComplete( unsigned int channel,
323  unsigned int sequenceNo,
324  void *userParam )
325 {
326  UARTDRV_Handle_t handle;
327  UARTDRV_Buffer_t *buffer;
328  (void)channel;
329  (void)sequenceNo;
330 
331  handle = (UARTDRV_Handle_t)userParam;
332  GetTailBuffer(handle->rxQueue, &buffer);
333 
334  if (handle->initData.port->IF & USART_IF_FERR)
335  {
336  buffer->transferStatus = ECODE_EMDRV_UARTDRV_FRAME_ERROR;
337  buffer->itemsRemaining = buffer->transferCount; // nothing received
338  handle->initData.port->IFC = USART_IFC_FERR;
339  }
340  else if (handle->initData.port->IF & USART_IF_PERR)
341  {
342  buffer->transferStatus = ECODE_EMDRV_UARTDRV_PARITY_ERROR;
343  buffer->itemsRemaining = buffer->transferCount; // nothing received
344  handle->initData.port->IFC = USART_IFC_PERR;
345  }
346  else
347  {
348  buffer->transferStatus = ECODE_EMDRV_UARTDRV_OK;
349  buffer->itemsRemaining = 0;
350  }
351 
352  INT_Disable();
353 
354  if (buffer->callback != NULL)
355  {
356  buffer->callback(handle, buffer->transferStatus, buffer->data, buffer->transferCount - buffer->itemsRemaining);
357  }
358  // Dequeue the current tail RX operation, check if more in queue
359  DequeueBuffer(handle->rxQueue, &buffer);
360 
361  if (handle->rxQueue->used > 0)
362  {
363  GetTailBuffer(handle->rxQueue, &buffer);
364  StartReceiveDma(handle, buffer);
365  }
366  else
367  {
368  #if (EMDRV_UARTDRV_HW_FLOW_CONTROL_ENABLE)
369  handle->fcSelfState = uartdrvFlowControlOff;
370  FcApplyState(handle);
371  #endif
372  handle->rxDmaActive = false;
373  if (handle->initData.fcType != uartdrvFlowControlHwUart)
374  {
375  DisableReceiver(handle);
376  }
377  }
378  INT_Enable();
379  return true;
