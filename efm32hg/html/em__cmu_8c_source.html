183 static void setHfLeConfig(uint32_t hfFreq, uint32_t maxLeFreq)
184 {
185  /* Check for 1 bit fields. BUS_RegBitWrite() below are going to fail if the
186  fields are changed to more than 1 bit. */
187  EFM_ASSERT((CMU_HFLE_WS_MASK >> CMU_HFLE_WS_SHIFT) == 0x1);
188  EFM_ASSERT((CMU_HFLE_PRESC_MASK >> CMU_HFLE_PRESC_SHIFT) == 0x1);
189 
190  /* 0: set 0 wait-states and DIV2
191  1: set 1 wait-states and DIV4 */
192  unsigned int val = (hfFreq <= maxLeFreq) ? 0 : 1;
193  BUS_RegBitWrite(&CMU->CTRL, CMU_HFLE_WS_SHIFT, val);
194  BUS_RegBitWrite(&CMU_HFLE_PRESC_REG, CMU_HFLE_PRESC_SHIFT, val);
195 }

335 static void flashWaitStateControl(uint32_t coreFreq)
336 {
337  uint32_t mode;
338  bool mscLocked;
339 #if defined( MSC_READCTRL_MODE_WS0SCBTP )
340  bool scbtpEn; /* Suppressed Conditional Branch Target Prefetch setting. */
341 #endif
342 
343  /* Make sure the MSC is unlocked */
344  mscLocked = MSC->LOCK;
345  MSC->LOCK = MSC_UNLOCK_CODE;
346 
347  /* Get mode and SCBTP enable */
348  mode = MSC->READCTRL & _MSC_READCTRL_MODE_MASK;
349 #if defined( MSC_READCTRL_MODE_WS0SCBTP )
350  switch(mode)
351  {
352  case MSC_READCTRL_MODE_WS0:
353  case MSC_READCTRL_MODE_WS1:
354 #if defined( MSC_READCTRL_MODE_WS2 )
355  case MSC_READCTRL_MODE_WS2:
356 #endif
357  scbtpEn = false;
358  break;
359 
360  default: /* WSxSCBTP */
361  scbtpEn = true;
362  break;
363  }
364 #endif
365 
366 
367  /* Set mode based on the core clock frequency and SCBTP enable */
368 #if defined( MSC_READCTRL_MODE_WS0SCBTP )
369  if (false)
370  {
371  }
372 #if defined( MSC_READCTRL_MODE_WS2 )
373  else if (coreFreq > CMU_MAX_FREQ_1WS)
374  {
375  mode = (scbtpEn ? MSC_READCTRL_MODE_WS2SCBTP : MSC_READCTRL_MODE_WS2);
376  }
377 #endif
378  else if ((coreFreq <= CMU_MAX_FREQ_1WS) && (coreFreq > CMU_MAX_FREQ_0WS))
379  {
380  mode = (scbtpEn ? MSC_READCTRL_MODE_WS1SCBTP : MSC_READCTRL_MODE_WS1);
381  }
382  else
383  {
384  mode = (scbtpEn ? MSC_READCTRL_MODE_WS0SCBTP : MSC_READCTRL_MODE_WS0);
385  }
386 
387 #else /* If MODE and SCBTP is in separate register fields */
388 
389  if (false)
390  {
391  }
392 #if defined( MSC_READCTRL_MODE_WS2 )
393  else if (coreFreq > CMU_MAX_FREQ_1WS)
394  {
395  mode = MSC_READCTRL_MODE_WS2;
396  }
397 #endif
398  else if ((coreFreq <= CMU_MAX_FREQ_1WS) && (coreFreq > CMU_MAX_FREQ_0WS))
399  {
400  mode = MSC_READCTRL_MODE_WS1;
401  }
402  else
403  {
404  mode = MSC_READCTRL_MODE_WS0;
405  }
406 #endif
407 
408  /* BUS_RegMaskedWrite cannot be used here as it would temporarely set the
409  mode field to WS0 */
410  MSC->READCTRL = (MSC->READCTRL &~_MSC_READCTRL_MODE_MASK) | mode;
411 
412  if (mscLocked)
413  {
414  MSC->LOCK = 0;
415  }
416 }
