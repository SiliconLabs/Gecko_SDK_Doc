156 static void emuRestore(void)
157 {
158  uint32_t oscEnCmd;
159  uint32_t cmuLocked;
160 
161  /* Although we could use the CMU API for most of the below handling, we */
162  /* would like this function to be as efficient as possible. */
163 
164  /* CMU registers may be locked */
165  cmuLocked = CMU->LOCK & CMU_LOCK_LOCKKEY_LOCKED;
166  CMU_Unlock();
167 
168  /* AUXHFRCO are automatically disabled (except if using debugger). */
169  /* HFRCO, USHFRCO and HFXO are automatically disabled. */
170  /* LFRCO/LFXO may be disabled by SW in EM3. */
171  /* Restore according to status prior to entering energy mode. */
172  oscEnCmd = 0;
173  oscEnCmd |= ((cmuStatus & CMU_STATUS_HFRCOENS) ? CMU_OSCENCMD_HFRCOEN : 0);
174  oscEnCmd |= ((cmuStatus & CMU_STATUS_AUXHFRCOENS) ? CMU_OSCENCMD_AUXHFRCOEN : 0);
175  oscEnCmd |= ((cmuStatus & CMU_STATUS_LFRCOENS) ? CMU_OSCENCMD_LFRCOEN : 0);
176  oscEnCmd |= ((cmuStatus & CMU_STATUS_HFXOENS) ? CMU_OSCENCMD_HFXOEN : 0);
177  oscEnCmd |= ((cmuStatus & CMU_STATUS_LFXOENS) ? CMU_OSCENCMD_LFXOEN : 0);
178 #if defined( _CMU_STATUS_USHFRCOENS_MASK )
179  oscEnCmd |= ((cmuStatus & CMU_STATUS_USHFRCOENS) ? CMU_OSCENCMD_USHFRCOEN : 0);
180 #endif
181  CMU->OSCENCMD = oscEnCmd;
182 
183 
184 #if defined( _CMU_HFCLKSTATUS_RESETVALUE )
185  /* Restore oscillator used for clocking core */
186  switch (cmuHfclkStatus & _CMU_HFCLKSTATUS_SELECTED_MASK)
187  {
188  case CMU_HFCLKSTATUS_SELECTED_LFRCO:
189  /* HFRCO could only be selected if the autostart HFXO feature is not
190  * enabled, otherwise the HFXO would be started and selected automatically.
191  * Note: this error hook helps catching erroneous oscillator configurations,
192  * when the AUTOSTARTSELEM0EM1 is set in CMU_HFXOCTRL. */
193  if (!(CMU->HFXOCTRL & CMU_HFXOCTRL_AUTOSTARTSELEM0EM1))
194  {
195  /* Wait for LFRCO to stabilize */
196  while (!(CMU->STATUS & CMU_STATUS_LFRCORDY))
197  ;
198  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_LFRCO;
199  }
200  else
201  {
202  EFM_ASSERT(0);
203  }
204  break;
205 
206  case CMU_HFCLKSTATUS_SELECTED_LFXO:
207  /* Wait for LFXO to stabilize */
208  while (!(CMU->STATUS & CMU_STATUS_LFXORDY))
209  ;
210  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_LFXO;
211  break;
212 
213  case CMU_HFCLKSTATUS_SELECTED_HFXO:
214  /* Wait for HFXO to stabilize */
215  while (!(CMU->STATUS & CMU_STATUS_HFXORDY))
216  ;
217  CMU->HFCLKSEL = CMU_HFCLKSEL_HF_HFXO;
218  break;
219 
220  default: /* CMU_HFCLKSTATUS_SELECTED_HFRCO */
221  /* If core clock was HFRCO core clock, it is automatically restored to */
222  /* state prior to entering energy mode. No need for further action. */
223  break;
224  }
225 #else
226  switch (cmuStatus & (CMU_STATUS_HFRCOSEL
227  | CMU_STATUS_HFXOSEL
228  | CMU_STATUS_LFRCOSEL
229 #if defined( CMU_STATUS_USHFRCODIV2SEL )
230  | CMU_STATUS_USHFRCODIV2SEL
231 #endif
232  | CMU_STATUS_LFXOSEL))
233  {
234  case CMU_STATUS_LFRCOSEL:
235  /* Wait for LFRCO to stabilize */
236  while (!(CMU->STATUS & CMU_STATUS_LFRCORDY))
237  ;
238  CMU->CMD = CMU_CMD_HFCLKSEL_LFRCO;
239  break;
240 
241  case CMU_STATUS_LFXOSEL:
242  /* Wait for LFXO to stabilize */
243  while (!(CMU->STATUS & CMU_STATUS_LFXORDY))
244  ;
245  CMU->CMD = CMU_CMD_HFCLKSEL_LFXO;
246  break;
247 
248  case CMU_STATUS_HFXOSEL:
249  /* Wait for HFXO to stabilize */
250  while (!(CMU->STATUS & CMU_STATUS_HFXORDY))
251  ;
252  CMU->CMD = CMU_CMD_HFCLKSEL_HFXO;
253  break;
254 
255 #if defined( CMU_STATUS_USHFRCODIV2SEL )
256  case CMU_STATUS_USHFRCODIV2SEL:
257  /* Wait for USHFRCO to stabilize */
258  while (!(CMU->STATUS & CMU_STATUS_USHFRCORDY))
259  ;
260  CMU->CMD = _CMU_CMD_HFCLKSEL_USHFRCODIV2;
261  break;
262 #endif
263 
264  default: /* CMU_STATUS_HFRCOSEL */
265  /* If core clock was HFRCO core clock, it is automatically restored to */
266  /* state prior to entering energy mode. No need for further action. */
267  break;
268  }
269 
270  /* If HFRCO was disabled before entering Energy Mode, turn it off again */
271  /* as it is automatically enabled by wake up */
272  if ( ! (cmuStatus & CMU_STATUS_HFRCOENS) )
273  {
274  CMU->OSCENCMD = CMU_OSCENCMD_HFRCODIS;
275  }
276 #endif
277  /* Restore CMU register locking */
278  if (cmuLocked)
279  {
280  CMU_Lock();
281  }
282 }

1064 static void currentLimitersUpdate(void)
1065 {
1066  uint32_t lncLimSel;
1067  uint32_t zdetLimSel;
1068  uint32_t pFetCnt;
1069  uint16_t maxReverseCurrent_mA;
1070 
1071  /* 80mA as recommended peak in Application Note AN0948.
1072  The peak current is the average current plus 50% of the current ripple.
1073  Hence, a 14mA average current is recommended in LP mode. Since LP PFETCNT is also
1074  a constant, we get lpcLimImSel = 1. The following calculation is provided
1075  for documentation only. */
1076  const uint32_t lpcLim = (((14 + 40) + ((14 + 40) / 2))
1077  / (5 * (DCDC_LP_PFET_CNT + 1)))
1078  - 1;
1079  const uint32_t lpcLimSel = lpcLim << _EMU_DCDCMISCCTRL_LPCLIMILIMSEL_SHIFT;
1080 
1081  /* Get enabled PFETs */
1082  pFetCnt = (EMU->DCDCMISCCTRL & _EMU_DCDCMISCCTRL_PFETCNT_MASK)
1083  >> _EMU_DCDCMISCCTRL_PFETCNT_SHIFT;
1084 
1085  /* Compute LN current limiter threshold from nominal user input current and
1086  LN PFETCNT as described in the register description for
1087  EMU_DCDCMISCCTRL_LNCLIMILIMSEL. */
1088  lncLimSel = (((dcdcMaxCurrent_mA + 40) + ((dcdcMaxCurrent_mA + 40) / 2))
1089  / (5 * (pFetCnt + 1)))
1090  - 1;
1091 
1092  /* Saturate the register field value */
1093  lncLimSel = SL_MIN(lncLimSel,
1094  _EMU_DCDCMISCCTRL_LNCLIMILIMSEL_MASK
1095  >> _EMU_DCDCMISCCTRL_LNCLIMILIMSEL_SHIFT);
1096 
1097  lncLimSel <<= _EMU_DCDCMISCCTRL_LNCLIMILIMSEL_SHIFT;
1098 
1099  /* Check for overflow */
1100  EFM_ASSERT((lncLimSel & ~_EMU_DCDCMISCCTRL_LNCLIMILIMSEL_MASK) == 0x0);
1101  EFM_ASSERT((lpcLimSel & ~_EMU_DCDCMISCCTRL_LPCLIMILIMSEL_MASK) == 0x0);
1102 
1103  EMU->DCDCMISCCTRL = (EMU->DCDCMISCCTRL & ~(_EMU_DCDCMISCCTRL_LNCLIMILIMSEL_MASK
1104  | _EMU_DCDCMISCCTRL_LPCLIMILIMSEL_MASK))
1105  | (lncLimSel | lpcLimSel);

1107

1115

1117  / ((2 * (pFetCnt + 1)) + ((pFetCnt + 1) / 2)) );
1118  /* Saturate the register field value */
1119  zdetLimSel = SL_MIN(zdetLimSel,
1120  _EMU_DCDCZDETCTRL_ZDETILIMSEL_MASK
1121  >> _EMU_DCDCZDETCTRL_ZDETILIMSEL_SHIFT);
1122 
1123  zdetLimSel <<= _EMU_DCDCZDETCTRL_ZDETILIMSEL_SHIFT;
1124 
1125  /* Check for overflow */
1126  EFM_ASSERT((zdetLimSel & ~_EMU_DCDCZDETCTRL_ZDETILIMSEL_MASK) == 0x0);
1127 
1128  EMU->DCDCZDETCTRL = (EMU->DCDCZDETCTRL & ~_EMU_DCDCZDETCTRL_ZDETILIMSEL_MASK)

1130  }
