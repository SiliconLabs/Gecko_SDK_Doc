617 /***************************************************************************/
620 void KSZ8851SNL_Init(void)
621 {
622  uint16_t data;
623 
624  /* Initialize SPI Interface */
625  KSZ8851SNL_SPI_Init();
626 
627  /* Reset PHY module */
628  KSZ8851SNL_SPI_WriteRegister(PHYRR, PHY_RESET);
629 
630  /* Reset Soft (clear registers of PHY, MAC, QMU, DMA) */
631  data = KSZ8851SNL_SPI_ReadRegister(GRR);
632  data &= ~GLOBAL_SOFT_RESET;
633  KSZ8851SNL_SPI_WriteRegister(GRR, data);
634 
635  /* Reset Soft (clear registers of PHY, MAC, QMU, DMA) */
636  data = KSZ8851SNL_SPI_ReadRegister(GRR);
637  data &= ~GLOBAL_SOFT_RESET;
638  KSZ8851SNL_SPI_WriteRegister(GRR, data);
639 
640  /* Read the chip ID and check if that is correct */
641  data = KSZ8851SNL_SPI_ReadRegister(CIDER);
642 
643  /* The CIDER lower bits [3..1] are defined as revision number,
644  * thus a mask needs to be applied
645  */
646 
647  /* this is a hard assert - abort */
648  EFM_ASSERT((data & CHIP_ID_MASK) == KSZ8851SNL_CHIP_ID);

658  KSZ8851SNL_SPI_WriteRegister(TXFDPR, FD_PTR_AUTO_INC);

661  * flow control,
662  * padding,
663  * CRC,
664  * checksum generation.
665  */
666  KSZ8851SNL_SPI_WriteRegister(TXCR, TX_FLOW_CTRL_CONFIG);

669  KSZ8851SNL_SPI_WriteRegister(RXFDPR, FD_PTR_AUTO_INC);
670

672  KSZ8851SNL_SPI_WriteRegister(RXFCTR, FRAME_COUNT_THRESHOLD);

691  */

730 /***************************************************************************/
737 bool KSZ8851SNL_TransmitBegin(uint16_t length)
738 {
739  uint16_t txmir;
740  uint16_t data, reqSize;
741  uint8_t outbuf[4];
742 
743  /* Wait for previous frame to finish before setting up a new one */
744  while (KSZ8851SNL_SPI_ReadRegister(TXQCR) & TXQ_ENQUEUE)
745  {
746  ;
747  }
748 
749  /* Make sure there is room for
750  *
751  * 4 bytes Control Word
752  * 4 bytes Byte Count
753  * n bytes Packet
754  * 4 bytes CRC
755  */
756  reqSize = length + 12;
757  txmir = KSZ8851SNL_SPI_ReadRegister(TXMIR) & TX_MEM_AVAIL_MASK;
758  LWIP_DEBUGF(NETIF_DEBUG, ("KSZ8851SNL_LongTransmitInit: txmir =%hu reqSize = %hu \n", txmir, reqSize));
759 
760  if (txmir < reqSize)
761  {
762  /* TXQ is out of memory */
763  LWIP_DEBUGF(NETIF_DEBUG | LWIP_DBG_LEVEL_WARNING, ("Not enough TXQ Memory, available=%u required=%u\n", txmir, reqSize));
764  return false;
765  }
766 
767  LWIP_DEBUGF(NETIF_DEBUG, ("KSZ8851SNL_LongTransmitInit: Memory available > txmir =%hu reqSize = %hu \n", txmir, reqSize));
768  /* Enable TXQ write access */
769  data = KSZ8851SNL_SPI_ReadRegister(RXQCR) | RXQ_START_DMA;
770  KSZ8851SNL_SPI_WriteRegister(RXQCR, data);
771 
772  /* Write frame ID, control word and byte count */
773  outbuf[0] = (frameId++ & FRAME_ID_MASK);
774  outbuf[1] = 0x80; /* TX_INT_on_COMPLETION */
775  outbuf[2] = length & LSB_MASK;
776  outbuf[3] = length >> MSB_POS;
777 
778  /* Start the SPI Transfer and send frame header */
779  KSZ8851SNL_SPI_WriteFifoBegin();
780  KSZ8851SNL_SPI_WriteFifo(4, outbuf);
781 
782  return true;
783 }

862 /***************************************************************************/
875 uint16_t KSZ8851SNL_Receive(uint16_t length, uint8_t *buffer)
876 {
877  uint16_t data;
878  uint16_t rxStatus;
879  uint16_t rxPacketLength;
880  uint16_t frameLength;
881  uint16_t bytesToRead;
882 
883  EFM_ASSERT(buffer != NULL);
884 
885  while (rxFrameCount > 0)
886  {
887  rxFrameCount--;
888  /* Read the received frame status */
889  rxStatus = KSZ8851SNL_SPI_ReadRegister(RXFHSR);
890 
891  /* Check the consistency of the frame */
892  if ((!(rxStatus & VALID_FRAME_MASK)) || (rxStatus & CHECKSUM_VALID_FRAME_MASK))
893  {
894  /* Issue the Release error frame command */
895  ReleaseIncosistentFrame();
896  /* continue to next frame */
897  continue;
898  }
899  else
900  {
901  /* Read the byte size of the received frame */
902  rxPacketLength = KSZ8851SNL_SPI_ReadRegister(RXFHBCR) & RX_BYTE_CNT_MASK;
903 
904  /* round to dword boundary */
905  bytesToRead = 4 * ((rxPacketLength + 3) >> 2);
906  LWIP_DEBUGF(NETIF_DEBUG, ("KSZ8851SNL_Receive: rxPacketLength=%u, bytesToRead=%u \n", rxPacketLength, bytesToRead));
907  if ((bytesToRead > length) || (rxPacketLength <= 4))
908  {
909  /* Issue the Release error frame command */
910  ReleaseIncosistentFrame();
911  /* continue to next frame */
912  continue;
913  }
914 
915  /* Set QMU RXQ frame pointer to start of packet data. Note
916  * that we skip the status word and byte count since we
917  * already know this from RXFHSR and RXFHBCR.
918  */
919  KSZ8851SNL_SPI_WriteRegister(RXFDPR, 0x0004 | FD_PTR_AUTO_INC);
920 
921  /* Start QMU DMA transfer */
922  data = KSZ8851SNL_SPI_ReadRegister(RXQCR) | RXQ_START_DMA;
923  KSZ8851SNL_SPI_WriteRegister(RXQCR, data);
924 
925  /* Read the whole ethernet frame */
926  KSZ8851SNL_SPI_ReadFifo(bytesToRead, buffer);
927 
928  /* Stop QMU DMA transfer */
929  data = KSZ8851SNL_SPI_ReadRegister(RXQCR) & (~RXQ_START_DMA);
930  KSZ8851SNL_SPI_WriteRegister(RXQCR, data);
931 
932  /* Return frame length without CRC */
933  frameLength = rxPacketLength - 4;
934  return frameLength;
935  }
936  }
937  return 0;
938 }
