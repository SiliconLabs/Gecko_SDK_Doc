139 static void errataCallback(int8_t temp, TEMPDRV_LimitType_t limit)
140 {
141  volatile uint32_t *reg;
142  if (limit == TEMPDRV_LIMIT_HIGH)
143  {
144  errataState++;
145  tempdrvHighCallbacks[0] = errataHighTemp[errataState];
146  tempdrvLowCallbacks[0] = errataLowTemp[errataState];
147  }
148  else if (limit == TEMPDRV_LIMIT_LOW)
149  {
150  errataState--;
151  tempdrvHighCallbacks[0] = errataHighTemp[errataState];
152  tempdrvLowCallbacks[0] = errataLowTemp[errataState];
153  }
154  bool lock = false;
155  if (EMU->LOCK == EMU_LOCK_LOCKKEY_LOCKED)
156  {
157  lock = true;
158  EMU->LOCK = EMU_LOCK_LOCKKEY_UNLOCK;
159  }
160  reg = (volatile uint32_t *) 0x400E3190;
161  *reg = 0xADE8;
162  switch (errataState)
163  {
164  case ERRATA_LOW:
165  case ERRATA_MIDLOW:
166  case ERRATA_MIDHIGH:
167  reg = (volatile uint32_t *) 0x400E3164;
168  *reg |= 0x70;
169  reg = (volatile uint32_t *) 0x400E319C;
170  *reg |= 0x8;
171  reg = (volatile uint32_t *) 0x400E403C;
172  *reg &= ~(0xC00);
173  *reg |= (errataState & 0x2) << 10;
174  reg = (volatile uint32_t *) 0x400E319C;
175  *reg &= ~(0x8);
176  break;
177  case ERRATA_HIGH:
178  default:
179  reg = (volatile uint32_t *) 0x400E3164;
180  *reg &= ~(0x70);
181  break;
182  }
183  reg = (volatile uint32_t *) 0x400E3190;
184  *reg = 0;
185  if (lock == true)
186  {
187  EMU->LOCK = EMU_LOCK_LOCKKEY_LOCK;
188  }
189 }

409 static void updateInterrupts()
410 {
411  // Find lowest temperature active high callback
412  uint8_t index;
413  for (index = 0; index < TEMPDRV_CALLBACK_DEPTH; index++)
414  {
415  // filter out only entries with valid callbacks
416  if (tempdrvHighCallbacks[index].callback!=NULL)
417  {
418  if (highCallback->callback == NULL)
419  {
420  highCallback = &tempdrvHighCallbacks[index];
421  }
422  else
423  {
424  if (tempdrvHighCallbacks[index].temp>highCallback->temp)
425  {
426  highCallback = &tempdrvHighCallbacks[index];
427  }
428  }
429  }
430  }
431  // If active callback, set and enable interrupt
432  if (highCallback->callback!=NULL)
433  {
434  // EMU and TEMP are inversely proportional
435  EMU->TEMPLIMITS &= ~_EMU_TEMPLIMITS_TEMPLOW_MASK;
436  EMU->TEMPLIMITS |= highCallback->temp << _EMU_TEMPLIMITS_TEMPLOW_SHIFT;
437  EMU_IntEnable (EMU_IEN_TEMPLOW);
438  }
439  else
440  {
441  EMU_IntDisable (EMU_IEN_TEMPLOW);
442  }
443 
444  // Find highest temperature active low callback
445  for (index = 0; index < TEMPDRV_CALLBACK_DEPTH; index++)
446  {
447  // filter out only entries with valid callbacks
448  if (tempdrvLowCallbacks[index].callback!=NULL)
449  {
450  if (lowCallback->callback == NULL)
451  {
452  lowCallback = &tempdrvLowCallbacks[index];
453  }
454  else
455  {
456  if (tempdrvLowCallbacks[index].temp<lowCallback->temp)
457  {
458  lowCallback = &tempdrvLowCallbacks[index];
459  }
460  }
461  }
462  }
463  // If active callback, set and enable interrupt
464  if (lowCallback->callback!=NULL)
465  {
466  // EMU and TEMP are inversely proportional
467  EMU->TEMPLIMITS &= ~_EMU_TEMPLIMITS_TEMPHIGH_MASK;
468  EMU->TEMPLIMITS |= lowCallback->temp << _EMU_TEMPLIMITS_TEMPHIGH_SHIFT;
469  EMU_IntEnable (EMU_IEN_TEMPHIGH);
470  }
471  else
472  {
473  EMU_IntDisable (EMU_IEN_TEMPHIGH);
474  }
475 }

485 Ecode_t TEMPDRV_Init()
486 {
487  uint32_t DItemp, DIemu;
488 
489  // Flag up
490  TEMPDRV_InitState = true;
491 
492  // reset stack state by erasing callbacks
493  memset (tempdrvHighCallbacks, 0, sizeof (TEMPDRV_CallbackSet_t) * TEMPDRV_CALLBACK_DEPTH);
494  memset (tempdrvLowCallbacks, 0, sizeof (TEMPDRV_CallbackSet_t) * TEMPDRV_CALLBACK_DEPTH);
495  highCallback = &nullCallback;
496  lowCallback = &nullCallback;
497 
498  // Retrieve calibration data from DI page
499  DItemp = ((DEVINFO->CAL & _DEVINFO_CAL_TEMP_MASK) >> _DEVINFO_CAL_TEMP_SHIFT);
500  DIemu = ((DEVINFO->EMUTEMP & _DEVINFO_EMUTEMP_EMUTEMPROOM_MASK) >> _DEVINFO_EMUTEMP_EMUTEMPROOM_SHIFT);
501 
502  if ((DItemp == 0xFFFFFFFF ) || (DIemu == 0xFFFFFFFF))
503  {
504  // Missing DI page calibration data, substitute fixed values
505  DItemp = fallbackTEMP;
506  DIemu = fallbackEMU;
507 
508  }
509 
510  // calculate conversion offsets. Based on assumed slope of 5/8
511  calibrationEMU = (DIemu) + ((5*(DItemp))/8);
512  calibrationTEMP = (DItemp) + (8*(DIemu)/5);
513 
514  errataInit();
515 
516  disableInterrupts();
517  NVIC_ClearPendingIRQ(EMU_IRQn);
518  NVIC_EnableIRQ(EMU_IRQn);
519  updateInterrupts();
520 
521  return ECODE_EMDRV_TEMPDRV_OK;
522 }

635 Ecode_t TEMPDRV_RegisterCallback(int8_t temp,
636  TEMPDRV_LimitType_t limit,
637  TEMPDRV_Callback_t callback)
638 {
639  TEMPDRV_CallbackSet_t *set;
640  if (TEMPDRV_InitState == false)
641  {
642  return ECODE_EMDRV_TEMPDRV_NO_INIT;
643  }
644  // cannot register null callback
645  if (callback == NULL)
646  {
647  return ECODE_EMDRV_TEMPDRV_PARAM_ERROR;
648  }
649  if (limit == TEMPDRV_LIMIT_HIGH)
650  {
651  // current temperature is already higher than requested temperature
652  if (TEMPDRV_GetTemp() > temp)
653  {
654  return ECODE_EMDRV_TEMPDRV_BAD_LIMIT;
655  }
656  // Define callback set
657  set = tempdrvHighCallbacks;
658  }
659  else if (limit == TEMPDRV_LIMIT_LOW)
660  {
661  // current temperature is already lower than requested temperature
662  if (TEMPDRV_GetTemp() < temp)
663  {
664  return ECODE_EMDRV_TEMPDRV_BAD_LIMIT;
665  }
666  // Define callback set
667  set = tempdrvLowCallbacks;
668  }
669  else
670  {
671  // Invalid limit
672  return ECODE_EMDRV_TEMPDRV_PARAM_ERROR;
673  }
674 
675  // Cannot register duplicate temperature callback
676  if (checkForDuplicates(set, temp) == true)
677  {
678  return ECODE_EMDRV_TEMPDRV_DUP_TEMP;
679  }
680 
681  return addCallback(set, temp, callback);
682 }
