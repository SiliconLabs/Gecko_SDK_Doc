176 void GPIO_ExtIntConfig(GPIO_Port_TypeDef port,
177  unsigned int pin,
178  unsigned int intNo,
179  bool risingEdge,
180  bool fallingEdge,
181  bool enable)
182 {
183  uint32_t tmp;
184 #if !defined(_GPIO_EXTIPINSELL_MASK)
185  (void)intNo;
186 #endif
187 
188  EFM_ASSERT(GPIO_PORT_PIN_VALID(port, pin));
189 #if defined(_GPIO_EXTIPINSELL_MASK)
190  EFM_ASSERT(GPIO_INTNO_PIN_VALID(intNo, pin));
191 #endif
192 
193  /* There are two registers controlling the interrupt configuration:
194  * The EXTIPSELL register controls pins 0-7 and EXTIPSELH controls
195  * pins 8-15. */
196  if (pin < 8)
197  {
198  BUS_RegMaskedWrite(&GPIO->EXTIPSELL,
199  _GPIO_EXTIPSELL_EXTIPSEL0_MASK
200  << (_GPIO_EXTIPSELL_EXTIPSEL1_SHIFT * pin),
201  port << (_GPIO_EXTIPSELL_EXTIPSEL1_SHIFT * pin));
202  }
203  else
204  {
205  tmp = pin - 8;
206  BUS_RegMaskedWrite(&GPIO->EXTIPSELH,
207  _GPIO_EXTIPSELH_EXTIPSEL8_MASK
208  << (_GPIO_EXTIPSELH_EXTIPSEL9_SHIFT * tmp),
209  port << (_GPIO_EXTIPSELH_EXTIPSEL9_SHIFT * tmp));
210  }
211 
212 #if defined(_GPIO_EXTIPINSELL_MASK)
213  /* There are two registers controlling the interrupt/pin number mapping:
214  * The EXTIPINSELL register controls interrupt 0-7 and EXTIPINSELH controls
215  * interrupt 8-15. */
216  if (intNo < 8)
217  {
218  BUS_RegMaskedWrite(&GPIO->EXTIPINSELL,
219  _GPIO_EXTIPINSELL_EXTIPINSEL0_MASK
220  << (_GPIO_EXTIPINSELL_EXTIPINSEL1_SHIFT * intNo),
221  (pin & _GPIO_EXTIPINSELL_EXTIPINSEL0_MASK)
222  << (_GPIO_EXTIPINSELL_EXTIPINSEL1_SHIFT * intNo));
223  }
224  else
225  {
226  tmp = intNo - 8;
227  BUS_RegMaskedWrite(&GPIO->EXTIPINSELH,
228  _GPIO_EXTIPINSELH_EXTIPINSEL8_MASK
229  << (_GPIO_EXTIPINSELH_EXTIPINSEL9_SHIFT * tmp),
230  (pin & _GPIO_EXTIPINSELH_EXTIPINSEL8_MASK)
231  << (_GPIO_EXTIPSELH_EXTIPSEL9_SHIFT * tmp));
232  }
233 #endif
234 
235  /* Enable/disable rising edge */
236  BUS_RegBitWrite(&(GPIO->EXTIRISE), pin, risingEdge);
237 
238  /* Enable/disable falling edge */
239  BUS_RegBitWrite(&(GPIO->EXTIFALL), pin, fallingEdge);
240 
241  /* Clear any pending interrupt */
242  GPIO->IFC = 1 << pin;
243 
244  /* Finally enable/disable interrupt */
245  BUS_RegBitWrite(&(GPIO->IEN), pin, enable);
246 }

265 void GPIO_PinModeSet(GPIO_Port_TypeDef port,
266  unsigned int pin,
267  GPIO_Mode_TypeDef mode,
268  unsigned int out)
269 {
270  EFM_ASSERT(GPIO_PORT_PIN_VALID(port, pin));
271 
272  /* If disabling pin, do not modify DOUT in order to reduce chance for */
273  /* glitch/spike (may not be sufficient precaution in all use cases) */
274  if (mode != gpioModeDisabled)
275  {
276  if (out)
277  {
278  GPIO_PinOutSet(port, pin);
279  }
280  else
281  {
282  GPIO_PinOutClear(port, pin);
283  }
284  }
285 
286  /* There are two registers controlling the pins for each port. The MODEL
287  * register controls pins 0-7 and MODEH controls pins 8-15. */
288  if (pin < 8)
289  {
290  GPIO->P[port].MODEL = (GPIO->P[port].MODEL & ~(0xF << (pin * 4)))
291  | (mode << (pin * 4));
292  }
293  else
294  {
295  GPIO->P[port].MODEH = (GPIO->P[port].MODEH & ~(0xF << ((pin - 8) * 4)))
296  | (mode << ((pin - 8) * 4));
297  }
298 
299  if (mode == gpioModeDisabled)
300  {
301  if (out)
302  {
303  GPIO_PinOutSet(port, pin);
304  }
305  else
306  {
307  GPIO_PinOutClear(port, pin);
308  }
309  }
310 }
