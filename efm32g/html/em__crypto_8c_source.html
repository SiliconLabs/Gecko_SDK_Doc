404 void CRYPTO_SHA_1(CRYPTO_TypeDef * crypto,
405  const uint8_t * msg,
406  uint64_t msgLen,
407  CRYPTO_SHA1_Digest_TypeDef msgDigest)
408 {
409  uint32_t temp;
410  int len;
411  int blockLen;
412  uint32_t shaBlock[CRYPTO_SHA1_BLOCK_SIZE_IN_32BIT_WORDS]=
413  {
414  /* Initial value */
415  0x67452301, 0xefcdab89, 0x98badcfe, 0x10325476, 0xc3d2e1f0
416  };
417  uint8_t * p8ShaBlock = (uint8_t *) shaBlock;
418 
419  /* Initialize crypto module to do SHA-1. */
420  crypto->CTRL = CRYPTO_CTRL_SHA_SHA1;
421  crypto->SEQCTRL = 0;
422  crypto->SEQCTRLB = 0;
423 
424  /* Set result width of MADD32 operation. */
425  CRYPTO_ResultWidthSet(crypto, cryptoResult256Bits);
426 
427  /* Write init value to DDATA1. */
428  CRYPTO_DDataWrite(&crypto->DDATA1, shaBlock);
429 
430  /* Copy data to DDATA0 and select DDATA0 and DDATA1 for SHA operation. */
431  CRYPTO_EXECUTE_2(crypto,
432  CRYPTO_CMD_INSTR_DDATA1TODDATA0,
433  CRYPTO_CMD_INSTR_SELDDATA0DDATA1);
434 
435  len = msgLen;
436 
437  while (len >= CRYPTO_SHA1_BLOCK_SIZE_IN_BYTES)
438  {
439  /* Write block to QDATA1. */
440  CRYPTO_QDataWrite(&crypto->QDATA1BIG, (uint32_t *) msg);
441 
442  /* Execute SHA */
443  CRYPTO_EXECUTE_3(crypto,
444  CRYPTO_CMD_INSTR_SHA,
445  CRYPTO_CMD_INSTR_MADD32,
446  CRYPTO_CMD_INSTR_DDATA0TODDATA1);
447 
448  len -= CRYPTO_SHA1_BLOCK_SIZE_IN_BYTES;
449  msg += CRYPTO_SHA1_BLOCK_SIZE_IN_BYTES;
450  }
451 
452  blockLen = 0;
453 
454  /* Build the last (or second to last) block */
455  for (; len; len--)
456  p8ShaBlock[blockLen++] = *msg++;
457 
458  /* append the '1' bit */
459  p8ShaBlock[blockLen++] = 0x80;
460 
461  /* if the length is currently above 56 bytes we append zeros
462  * then compress. Then we can fall back to padding zeros and length
463  * encoding like normal.
464  */
465  if (blockLen > 56)
466  {
467  while (blockLen < 64)
468  p8ShaBlock[blockLen++] = 0;
469 
470  /* Write block to QDATA1BIG. */
471  CRYPTO_QDataWrite(&crypto->QDATA1BIG, shaBlock);
472 
473  /* Execute SHA */
474  CRYPTO_EXECUTE_3(crypto,
475  CRYPTO_CMD_INSTR_SHA,
476  CRYPTO_CMD_INSTR_MADD32,
477  CRYPTO_CMD_INSTR_DDATA0TODDATA1);
478  blockLen = 0;
479  }
480 
481  /* pad upto 56 bytes of zeroes */
482  while (blockLen < 56)
483  p8ShaBlock[blockLen++] = 0;
484 
485  /* And finally, encode the message length. */
486  {
487  uint64_t msgLenInBits = msgLen << 3;
488  temp = msgLenInBits >> 32;
489  *(uint32_t*)&p8ShaBlock[56] = SWAP32(temp);
490  temp = msgLenInBits & 0xFFFFFFFF;
491  *(uint32_t*)&p8ShaBlock[60] = SWAP32(temp);
492  }
493 
494  /* Write block to QDATA1BIG. */
495  CRYPTO_QDataWrite(&crypto->QDATA1BIG, shaBlock);
496 
497  /* Execute SHA */
498  CRYPTO_EXECUTE_3(crypto,
499  CRYPTO_CMD_INSTR_SHA,
500  CRYPTO_CMD_INSTR_MADD32,
501  CRYPTO_CMD_INSTR_DDATA0TODDATA1);
502 
503  /* Read resulting message digest from DDATA0BIG. */
504  ((uint32_t*)msgDigest)[0] = crypto->DDATA0BIG;
505  ((uint32_t*)msgDigest)[1] = crypto->DDATA0BIG;
506  ((uint32_t*)msgDigest)[2] = crypto->DDATA0BIG;
507  ((uint32_t*)msgDigest)[3] = crypto->DDATA0BIG;
508  ((uint32_t*)msgDigest)[4] = crypto->DDATA0BIG;
509  temp = crypto->DDATA0BIG;
510  temp = crypto->DDATA0BIG;
511  temp = crypto->DDATA0BIG;
512 }

685 void CRYPTO_Mul(CRYPTO_TypeDef * crypto,
686  uint32_t * A, int aSize,
687  uint32_t * B, int bSize,
688  uint32_t * R, int rSize)
689 {
690  int i, j;
691 
692  /**************** Initializations ******************/
693 
694 #ifdef USE_VARIABLE_SIZED_DATA_LOADS
695  int numWordsLastOperandA = (aSize&PARTIAL_OPERAND_WIDTH_MASK)>>5;
696  int numPartialOperandsA = numWordsLastOperandA ?
697  (aSize >> PARTIAL_OPERAND_WIDTH_LOG2) + 1 :
698  aSize >> PARTIAL_OPERAND_WIDTH_LOG2;
699  int numWordsLastOperandB = (bSize&PARTIAL_OPERAND_WIDTH_MASK)>>5;
700  int numPartialOperandsB = numWordsLastOperandB ?
701  (bSize >> PARTIAL_OPERAND_WIDTH_LOG2) + 1 :
702  bSize >> PARTIAL_OPERAND_WIDTH_LOG2;
703  int numWordsLastOperandR = (rSize&PARTIAL_OPERAND_WIDTH_MASK)>>5;
704  int numPartialOperandsR = numWordsLastOperandR ?
705  (rSize >> PARTIAL_OPERAND_WIDTH_LOG2) + 1 :
706  rSize >> PARTIAL_OPERAND_WIDTH_LOG2;
707  EFM_ASSERT(numPartialOperandsA + numPartialOperandsB <= numPartialOperandsR);
708 #else
709  int numPartialOperandsA = aSize >> PARTIAL_OPERAND_WIDTH_LOG2;
710  int numPartialOperandsB = bSize >> PARTIAL_OPERAND_WIDTH_LOG2;
711  EFM_ASSERT((aSize & PARTIAL_OPERAND_WIDTH_MASK) == 0);
712  EFM_ASSERT((bSize & PARTIAL_OPERAND_WIDTH_MASK) == 0);
713 #endif
714  EFM_ASSERT(aSize + bSize <= rSize);
715 
716  /* Set R to zero. */
717  cryptoBigintZeroize(R, rSize >> 5);
718 
719  /* Set multiplication width. */
720  crypto->WAC = CRYPTO_WAC_MULWIDTH_MUL128 | CRYPTO_WAC_RESULTWIDTH_256BIT;
721 
722  /* Setup DMA request signalling in order for MCU to run in parallel with
723  CRYPTO instruction sequence execution, and prepare data loading which
724  can take place immediately when CRYPTO is ready inside the instruction
725  sequence. */
726  crypto->CTRL =
727  CRYPTO_CTRL_DMA0RSEL_DATA0 | CRYPTO_CTRL_DMA0MODE_FULL |
728  CRYPTO_CTRL_DMA1RSEL_DATA1 | CRYPTO_CTRL_DMA1MODE_FULL;
729 
730  CRYPTO_EXECUTE_4(crypto,
731  CRYPTO_CMD_INSTR_CCLR, /* Carry = 0 */
732  CRYPTO_CMD_INSTR_CLR, /* DDATA0 = 0 */
733  /* clear result accumulation register */
734  CRYPTO_CMD_INSTR_DDATA0TODDATA2,
735  CRYPTO_CMD_INSTR_SELDDATA1DDATA3);
736  /*
737  register map:
738  DDATA0: working register
739  DDATA1: B(j)
740  DDATA2: R(i+j+1) and R(i+j), combined with DMA entry for B(j)
741  DDATA3: A(i)
742  */
743 
744  CRYPTO_SEQ_LOAD_10(crypto,
745  /* Temporarily load partial operand B(j) to DATA0. */
746  /* R(i+j+1) is still in DATA1 */
747  CRYPTO_CMD_INSTR_DMA0TODATA,
748  /* Move B(j) to DDATA1 */
749  CRYPTO_CMD_INSTR_DDATA2TODDATA1,
750 
751  /* Restore previous partial result (now R(i+j)) */
752  CRYPTO_CMD_INSTR_DATA1TODATA0,
753 
754  /* Load next partial result R(i+j+1) */
755  CRYPTO_CMD_INSTR_DMA1TODATA,
756 
757  /* Execute partial multiplication A(i)inDDATA1 * B(j)inDDATA3*/
758  CRYPTO_CMD_INSTR_MULO,
759 
760  /* Add the result to the previous partial result */
761  /* AND take the previous carry value into account */
762  /* at the right place (bit 128, ADDIC instruction */
763  CRYPTO_CMD_INSTR_SELDDATA0DDATA2,
764  CRYPTO_CMD_INSTR_ADDIC,
765 
766  /* Save the new partial result (lower half) */
767  CRYPTO_CMD_INSTR_DDATA0TODDATA2,
768  CRYPTO_CMD_INSTR_DATATODMA0,
769  /* Reset the operand selector for next*/
770  CRYPTO_CMD_INSTR_SELDDATA2DDATA3
771  );
772 
773  /**************** End Initializations ******************/
774 
775  for(i=0; i<numPartialOperandsA; i++)
776  {
777  /* Load partial operand #1 A>>(i*PARTIAL_OPERAND_WIDTH) to DDATA1. */
778 #ifdef USE_VARIABLE_SIZED_DATA_LOADS
779  if ( (numWordsLastOperandA != 0) && ( i == numPartialOperandsA-1 ) )
780  CRYPTO_DataWriteVariableSize(&crypto->DATA2,
781  &A[i*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS],
782  numWordsLastOperandA);
783  else
784  CRYPTO_DataWrite(&crypto->DATA2, &A[i*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
785 #else
786  CRYPTO_DataWrite(&crypto->DATA2, &A[i*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
787 #endif
788 
789  /* Load partial result in R>>(i*PARTIAL_OPERAND_WIDTH) to DATA1. */
790 #ifdef USE_VARIABLE_SIZED_DATA_LOADS
791  if ( (numWordsLastOperandR != 0) && ( i == numPartialOperandsR-1 ) )
792  CRYPTO_DataWriteVariableSize(&crypto->DATA1,
793  &R[i*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS],
794  numWordsLastOperandR);
795  else
796  CRYPTO_DataWrite(&crypto->DATA1, &R[i*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
797 #else
798  CRYPTO_DataWrite(&crypto->DATA1, &R[i*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
799 #endif
800 
801  /* Clear carry */
802  crypto->CMD = CRYPTO_CMD_INSTR_CCLR;
803 
804  /* Setup number of sequence iterations and block size. */
805  crypto->SEQCTRL = CRYPTO_SEQCTRL_BLOCKSIZE_16BYTES
806  | (PARTIAL_OPERAND_WIDTH_IN_BYTES * numPartialOperandsB);
807 
808  /* Execute the MULtiply instruction sequence. */
809  CRYPTO_InstructionSequenceExecute(crypto);
810 
811  for (j=0; j<numPartialOperandsB; j++)
812  {
813  /* Load partial operand 2 B>>(j*`PARTIAL_OPERAND_WIDTH) to DDATA2
814  (via DATA0). */
815 #ifdef USE_VARIABLE_SIZED_DATA_LOADS
816  if ( (numWordsLastOperandB != 0) && ( j == numPartialOperandsB-1 ) )
817  CRYPTO_DataWriteVariableSize(&crypto->DATA0,
818  &B[j*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS],
819  numWordsLastOperandB);
820  else
821  CRYPTO_DataWrite(&crypto->DATA0,
822  &B[j*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
823 #else
824  CRYPTO_DataWrite(&crypto->DATA0,
825  &B[j*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
826 #endif
827 
828  /* Load most significant partial result
829  R>>((i+j+1)*`PARTIAL_OPERAND_WIDTH) into DATA1. */
830 #ifdef USE_VARIABLE_SIZED_DATA_LOADS
831  if ( (numWordsLastOperandR != 0) && ( (i+j+1) == numPartialOperandsR-1 ) )
832  CRYPTO_DataWriteVariableSize(&crypto->DATA1,
833  &R[(i+j+1)*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS],
834  numWordsLastOperandR);
835  else
836  CRYPTO_DataWrite(&crypto->DATA1,
837  &R[(i+j+1)*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
838 #else
839  CRYPTO_DataWrite(&crypto->DATA1,
840  &R[(i+j+1)*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
841 #endif
842  /* Store least significant partial result */
843  CRYPTO_DataRead(&crypto->DATA0,
844  &R[(i+j)*PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS]);
845 
846  } /* for (j=0; j<numPartialOperandsB; j++) */
847 
848  /* Handle carry at the end of the inner loop. */
849  if (CRYPTO_CarryIsSet(crypto))
850  cryptoBigintIncrement(&R[(i+numPartialOperandsB+1)
851  *PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS],
852  (numPartialOperandsA-i-1)
853  *PARTIAL_OPERAND_WIDTH_IN_32BIT_WORDS);
854 
855  CRYPTO_DataRead(&crypto->DATA1,
856  &R[(i+numPartialOperandsB)

1938 static inline void CRYPTO_AES_ProcessLoop(CRYPTO_TypeDef * crypto,
1939  uint32_t len,
1940  CRYPTO_DataReg_TypeDef inReg,
1941  uint32_t * in,
1942  CRYPTO_DataReg_TypeDef outReg,
1943  uint32_t * out)
1944 {
1945  len /= CRYPTO_AES_BLOCKSIZE;
1946  crypto->SEQCTRL = 16 << _CRYPTO_SEQCTRL_LENGTHA_SHIFT;
1947 
1948  while (len--)
1949  {
1950  /* Load data and trigger encryption */
1951  CRYPTO_DataWrite(inReg, (uint32_t *)in);
1952 
1953  crypto->CMD = CRYPTO_CMD_SEQSTART;
1954 
1955  /* Save encrypted/decrypted data */
1956  CRYPTO_DataRead(outReg, (uint32_t *)out);
1957 
1958  out += 4;
1959  in += 4;
1960  }
1961 }
