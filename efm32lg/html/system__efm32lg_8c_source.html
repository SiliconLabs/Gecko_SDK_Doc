157 /***************************************************************************/
167 uint32_t SystemHFClockGet(void)
168 {
169  uint32_t ret;
170 
171  switch (CMU->STATUS & (CMU_STATUS_HFRCOSEL | CMU_STATUS_HFXOSEL |
172  CMU_STATUS_LFRCOSEL | CMU_STATUS_LFXOSEL))
173  {
174  case CMU_STATUS_LFXOSEL:
175 #if (EFM32_LFXO_FREQ > 0)
176  ret = SystemLFXOClock;
177 #else
178  /* We should not get here, since core should not be clocked. May */
179  /* be caused by a misconfiguration though. */
180  ret = 0;
181 #endif
182  break;
183 
184  case CMU_STATUS_LFRCOSEL:
185  ret = EFM32_LFRCO_FREQ;
186  break;
187 
188  case CMU_STATUS_HFXOSEL:
189 #if (EFM32_HFXO_FREQ > 0)
190  ret = SystemHFXOClock;
191 #else
192  /* We should not get here, since core should not be clocked. May */
193  /* be caused by a misconfiguration though. */
194  ret = 0;
195 #endif
196  break;
197 
198  default: /* CMU_STATUS_HFRCOSEL */
199  switch (CMU->HFRCOCTRL & _CMU_HFRCOCTRL_BAND_MASK)
200  {
201  case CMU_HFRCOCTRL_BAND_28MHZ:
202  ret = 28000000;
203  break;
204 
205  case CMU_HFRCOCTRL_BAND_21MHZ:
206  ret = 21000000;
207  break;
208 
209  case CMU_HFRCOCTRL_BAND_14MHZ:
210  ret = 14000000;
211  break;
212 
213  case CMU_HFRCOCTRL_BAND_11MHZ:
214  ret = 11000000;
215  break;
216 
217  case CMU_HFRCOCTRL_BAND_7MHZ:
218  if ( GetProdRev() >= 19 )
219  ret = 6600000;
220  else
221  ret = 7000000;
222  break;
223 
224  case CMU_HFRCOCTRL_BAND_1MHZ:
225  if ( GetProdRev() >= 19 )
226  ret = 1200000;
227  else
228  ret = 1000000;
229  break;
230 
231  default:
232  ret = 0;
233  break;
234  }
235  break;
236  }
237 
238  return ret;
239 }
