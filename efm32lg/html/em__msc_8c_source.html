225 void MSC_Init(void)
226 {
227 #if defined( _MSC_TIMEBASE_MASK )
228  uint32_t freq, cycles;
229 #endif
230  /* Unlock the MSC */
231  MSC->LOCK = MSC_UNLOCK_CODE;
232  /* Disable writing to the flash */
233  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
234 
235  /* Call SystemCoreClockGet in order to set the global variable SystemCoreClock
236  which is used in MSC_LoadWriteData to make sure the frequency is
237  sufficiently high. If the clock frequency is changed then software is
238  responsible for calling MSC_Init or SystemCoreClockGet in order to set the
239  SystemCoreClock variable to the correct value. */
240  SystemCoreClockGet();
241 
242 #if defined( _MSC_TIMEBASE_MASK )
243  /* Configure MSC->TIMEBASE according to selected frequency */
244  freq = CMU_ClockFreqGet(cmuClock_AUX);
245 
246  /* Timebase 5us is used for the 1/1.2MHz band only. Note that the 1MHz band
247  is tuned to 1.2MHz on newer revisions. */
248  if (freq > 1200000)
249  {
250  /* Calculate number of clock cycles for 1us as base period */
251  freq = (freq * 11) / 10;
252  cycles = (freq / 1000000) + 1;
253 
254  /* Configure clock cycles for flash timing */
255  MSC->TIMEBASE = (MSC->TIMEBASE & ~(_MSC_TIMEBASE_BASE_MASK
256  | _MSC_TIMEBASE_PERIOD_MASK))
257  | MSC_TIMEBASE_PERIOD_1US
258  | (cycles << _MSC_TIMEBASE_BASE_SHIFT);
259  }
260  else
261  {
262  /* Calculate number of clock cycles for 5us as base period */
263  freq = (freq * 5 * 11) / 10;
264  cycles = (freq / 1000000) + 1;
265 
266  /* Configure clock cycles for flash timing */
267  MSC->TIMEBASE = (MSC->TIMEBASE & ~(_MSC_TIMEBASE_BASE_MASK
268  | _MSC_TIMEBASE_PERIOD_MASK))
269  | MSC_TIMEBASE_PERIOD_5US
270  | (cycles << _MSC_TIMEBASE_BASE_SHIFT);
271  }
272 #endif
273 }

364 MSC_Status_TypeDef MSC_LoadVerifyAddress(uint32_t* address)
365 {
366  uint32_t status;
367  uint32_t timeOut;
368 
369  /* Wait for the MSC to become ready. */
370  timeOut = MSC_PROGRAM_TIMEOUT;
371  while ((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
372  {
373  timeOut--;
374  }
375 
376  /* Check for timeout */
377  if (timeOut == 0)
378  {
379  return mscReturnTimeOut;
380  }
381  /* Load address */
382  MSC->ADDRB = (uint32_t)address;
383  MSC->WRITECMD = MSC_WRITECMD_LADDRIM;
384 
385  status = MSC->STATUS;
386  if (status & (MSC_STATUS_INVADDR | MSC_STATUS_LOCKED))
387  {
388  /* Check for invalid address */
389  if (status & MSC_STATUS_INVADDR)
390  return mscReturnInvalidAddr;
391  /* Check for write protected page */
392  if (status & MSC_STATUS_LOCKED)
393  return mscReturnLocked;
394  }
395  return mscReturnOk;
396 }

420 MSC_RAMFUNC_DEFINITION_BEGIN
421 MSC_Status_TypeDef MSC_LoadWriteData(uint32_t* data,
422  uint32_t numWords,
423  MSC_WriteStrategy_Typedef writeStrategy)
424 {
425  uint32_t timeOut;
426  uint32_t wordIndex;
427  uint32_t wordsPerDataPhase;
428  MSC_Status_TypeDef retval = mscReturnOk;
429 
430 #if defined(_MSC_WRITECTRL_LPWRITE_MASK) && defined(_MSC_WRITECTRL_WDOUBLE_MASK)
431  /* If LPWRITE (Low Power Write) is NOT enabled, set WDOUBLE (Write Double word) */
432  if (!(MSC->WRITECTRL & MSC_WRITECTRL_LPWRITE))
433  {
434  /* If the number of words to be written are odd, we need to align by writing
435  a single word first, before setting the WDOUBLE bit. */
436  if (numWords & 0x1)
437  {
438  /* Wait for the MSC to become ready for the next word. */
439  timeOut = MSC_PROGRAM_TIMEOUT;
440  while ((!(MSC->STATUS & MSC_STATUS_WDATAREADY)) && (timeOut != 0))
441  {
442  timeOut--;
443  }
444  /* Check for timeout */
445  if (timeOut == 0)
446  {
447  return mscReturnTimeOut;
448  }
449  /* Clear double word option, in order to write the initial single word. */
450  MSC->WRITECTRL &= ~MSC_WRITECTRL_WDOUBLE;
451  /* Write first data word. */
452  MSC->WDATA = *data++;
453  MSC->WRITECMD = MSC_WRITECMD_WRITEONCE;
454 
455  /* Wait for the operation to finish. It may be required to change the WDOUBLE
456  config after the initial write. It should not be changed while BUSY. */
457  timeOut = MSC_PROGRAM_TIMEOUT;
458  while((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
459  {
460  timeOut--;
461  }
462  /* Check for timeout */
463  if (timeOut == 0)
464  {
465  return mscReturnTimeOut;
466  }
467  /* Subtract this initial odd word for the write loop below */
468  numWords -= 1;
469  retval = mscReturnOk;
470  }
471  /* Now we can set the double word option in order to write two words per
472  data phase. */
473  MSC->WRITECTRL |= MSC_WRITECTRL_WDOUBLE;
474  wordsPerDataPhase = 2;
475  }
476  else
477 #endif /* defined( _MSC_WRITECTRL_LPWRITE_MASK ) && defined( _MSC_WRITECTRL_WDOUBLE_MASK ) */
478  {
479  wordsPerDataPhase = 1;
480  }
481 
482  /* Write the rest as double word write if wordsPerDataPhase == 2 */
483  if (numWords > 0)
484  {
485  /**** Write strategy: mscWriteIntSafe ****/
486  if (writeStrategy == mscWriteIntSafe)
487  {
488  /* Requires a system core clock at 1MHz or higher */
489  EFM_ASSERT(SystemCoreClock >= 1000000);
490  wordIndex = 0;
491  while(wordIndex < numWords)
492  {
493  MSC->WDATA = *data++;
494  wordIndex++;
495  if (wordsPerDataPhase == 2)
496  {
497  while (!(MSC->STATUS & MSC_STATUS_WDATAREADY));
498  MSC->WDATA = *data++;
499  wordIndex++;
500  }
501  MSC->WRITECMD = MSC_WRITECMD_WRITEONCE;
502 
503  /* Wait for the transaction to finish. */
504  timeOut = MSC_PROGRAM_TIMEOUT;
505  while ((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
506  {
507  timeOut--;
508  }
509  /* Check for timeout */
510  if (timeOut == 0)
511  {
512  retval = mscReturnTimeOut;
513  break;
514  }
515 #if defined( _EFM32_GECKO_FAMILY )
516  MSC->ADDRB += 4;
517  MSC->WRITECMD = MSC_WRITECMD_LADDRIM;
518 #endif
519  }
520  }
521 
522  /**** Write strategy: mscWriteFast ****/
523  else
524  {
525 #if defined( _EFM32_GECKO_FAMILY )
526  /* Gecko does not have auto-increment of ADDR. */
527  EFM_ASSERT(false);
528 #else
529  /* Requires a system core clock at 14MHz or higher. */
530  EFM_ASSERT(SystemCoreClock >= 14000000);
531 
532  wordIndex = 0;
533  mscRfIntDisable();
534  while(wordIndex < numWords)
535  {
536  /* Wait for the MSC to be ready for the next word. */
537  while (!(MSC->STATUS & MSC_STATUS_WDATAREADY))
538  {
539  /* If the write to MSC->WDATA below missed the 30us timeout and the
540  following MSC_WRITECMD_WRITETRIG command arrived while
541  MSC_STATUS_BUSY is 1, then the MSC_WRITECMD_WRITETRIG could be ignored by
542  the MSC. In this case, MSC_STATUS_WORDTIMEOUT is set to 1
543  and MSC_STATUS_BUSY is 0. A new trigger is therefore needed here to
544  complete write of data in MSC->WDATA.
545  If WDATAREADY became high since entry into this loop, exit and continue
546  to the next WDATA write.
547  */
548  if ((MSC->STATUS & (MSC_STATUS_WORDTIMEOUT
549  | MSC_STATUS_BUSY
550  | MSC_STATUS_WDATAREADY))
551  == MSC_STATUS_WORDTIMEOUT)
552  {
553  MSC->WRITECMD = MSC_WRITECMD_WRITETRIG;
554  }
555  }
556  MSC->WDATA = *data;
557  if ((wordsPerDataPhase == 1)
558  || ((wordsPerDataPhase == 2) && (wordIndex & 0x1)))
559  {
560  MSC->WRITECMD = MSC_WRITECMD_WRITETRIG;
561  }
562  data++;
563  wordIndex++;
564  }
565  mscRfIntEnable();
566 
567  /* Wait for the transaction to finish. */
568  timeOut = MSC_PROGRAM_TIMEOUT;
569  while ((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
570  {
571  timeOut--;
572  }
573  /* Check for timeout */
574  if (timeOut == 0)
575  {
576  retval = mscReturnTimeOut;
577  }
578 #endif
579  } /* writeStrategy */
580  }
581 
582 #if defined( _MSC_WRITECTRL_WDOUBLE_MASK )
583  /* Clear double word option, which should not be left on when returning. */
584  MSC->WRITECTRL &= ~MSC_WRITECTRL_WDOUBLE;
585 #endif
586 
587  return retval;
588 }

606 MSC_RAMFUNC_DEFINITION_BEGIN
607 MSC_Status_TypeDef MSC_WriteWordI(uint32_t *address,
608  void const *data,
609  uint32_t numBytes,
610  MSC_WriteStrategy_Typedef writeStrategy)
611 {
612  uint32_t wordCount;
613  uint32_t numWords;
614  uint32_t pageWords;
615  uint32_t* pData;
616  MSC_Status_TypeDef retval = mscReturnOk;
617 
618  /* Check alignment (Must be aligned to words) */
619  EFM_ASSERT(((uint32_t) address & 0x3) == 0);
620 
621  /* Check number of bytes. Must be divisable by four */
622  EFM_ASSERT((numBytes & 0x3) == 0);
623 
624  /* Enable writing to the MSC */
625  MSC->WRITECTRL |= MSC_WRITECTRL_WREN;
626 
627  /* Convert bytes to words */
628  numWords = numBytes >> 2;
629  EFM_ASSERT(numWords > 0);
630 
631  /* The following loop splits the data into chunks corresponding to flash pages.
632  The address is loaded only once per page, because the hardware automatically
633  increments the address internally for each data load inside a page. */
634  for (wordCount = 0, pData = (uint32_t *)data; wordCount < numWords; )
635  {
636  /* First we load address. The address is auto-incremented within a page.
637  Therefore the address phase is only needed once for each page. */
638  retval = MSC_LoadVerifyAddress(address + wordCount);
639  if (mscReturnOk != retval)
640  {
641  return retval;
642  }
643  /* Compute the number of words to write to the current page. */
644  pageWords =
645  (FLASH_PAGE_SIZE -
646  (((uint32_t) (address + wordCount)) & (FLASH_PAGE_SIZE - 1)))
647  / sizeof(uint32_t);
648  if (pageWords > numWords - wordCount)
649  {
650  pageWords = numWords - wordCount;
651  }
652  /* Now write the data in the current page. */
653  retval = MSC_LoadWriteData(pData, pageWords, writeStrategy);
654  if (mscReturnOk != retval)
655  {
656  break;
657  }
658  wordCount += pageWords;
659  pData += pageWords;
660  }
661 
662  /* Disable writing to the MSC */
663  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
664 
665 #if defined( _MSC_WRITECTRL_WDOUBLE_MASK )
666 #if ( WORDS_PER_DATA_PHASE == 2 )
667  /* Turn off double word write cycle support. */
668  MSC->WRITECTRL &= ~MSC_WRITECTRL_WDOUBLE;
669 #endif
670 #endif
671 
672  return retval;
673 }

706 {
707  uint32_t timeOut = MSC_PROGRAM_TIMEOUT;
708 
709  /* Address must be aligned to pages */
710  EFM_ASSERT((((uint32_t) startAddress) & (FLASH_PAGE_SIZE - 1)) == 0);
711 
712  /* Enable writing to the MSC */
713  MSC->WRITECTRL |= MSC_WRITECTRL_WREN;
714 
715  /* Load address */
716  MSC->ADDRB = (uint32_t)startAddress;
717  MSC->WRITECMD = MSC_WRITECMD_LADDRIM;
718 
719  /* Check for invalid address */
720  if (MSC->STATUS & MSC_STATUS_INVADDR)
721  {
722  /* Disable writing to the MSC */
723  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
724  return mscReturnInvalidAddr;
725  }
726  /* Check for write protected page */
727  if (MSC->STATUS & MSC_STATUS_LOCKED)
728  {
729  /* Disable writing to the MSC */
730  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
731  return mscReturnLocked;
732  }
733  /* Send erase page command */
734  MSC->WRITECMD = MSC_WRITECMD_ERASEPAGE;
735 
736  /* Wait for the erase to complete */
737  while ((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
738  {
739  timeOut--;
740  }
741  if (timeOut == 0)
742  {
743  /* Disable writing to the MSC */
744  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
745  return mscReturnTimeOut;
746  }
747  /* Disable writing to the MSC */
748  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
749  return mscReturnOk;
