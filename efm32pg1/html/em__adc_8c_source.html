449 uint32_t ADC_ScanSingleEndedInputAdd(ADC_InitScan_TypeDef *scanInit,
450  ADC_ScanInputGroup_TypeDef inputGroup,
451  ADC_PosSel_TypeDef singleEndedSel)
452 {
453  uint32_t currentSel;
454  uint32_t newSel;
455  uint32_t scanId;
456 
457  scanInit->diff = false;
458 
459  /* Check for unsupported APORTs */
460  EFM_ASSERT((singleEndedSel <= adcPosSelAPORT0YCH0) || (singleEndedSel >= adcPosSelAPORT0YCH15));
461 
462  /* Decode the input group select by shifting right by 3 */
463  newSel = singleEndedSel >> 3;
464 
465  currentSel = (scanInit->scanInputConfig.scanInputSel >> (inputGroup * 8)) & 0xFF;
466 
467  /* If none selected */
468  if (currentSel == 0xFF)
469  {
470  scanInit->scanInputConfig.scanInputSel &= ~(0xFF << (inputGroup * 8));
471  scanInit->scanInputConfig.scanInputSel |= (newSel << (inputGroup * 8));
472  }
473  else if (currentSel == newSel)
474  {
475  /* Ok, but do nothing. */
476  }
477  else
478  {
479  /* Invalid channel range. A range is already selected for this group. */
480  EFM_ASSERT(false);
481  }
482 
483  /* Update and return scan input enable mask (SCANMASK) */
484  scanId = (inputGroup * 8) + (singleEndedSel & 0x7);
485  EFM_ASSERT(scanId < 32);
486  scanInit->scanInputConfig.scanInputEn |= 0x1 << scanId;
487  return scanId;
488 }
