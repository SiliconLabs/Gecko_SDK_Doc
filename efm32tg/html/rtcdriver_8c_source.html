514 Ecode_t RTCDRV_StartTimer( RTCDRV_TimerID_t id,
515  RTCDRV_TimerType_t type,
516  uint32_t timeout,
517  RTCDRV_Callback_t callback,
518  void *user )
519 {
520  uint32_t timeElapsed, cnt, compVal, loopCnt = 0;
521  uint32_t timeToNextTimerCompletion;
522 
523  // Check if valid timer ID.
524  if ( id >= EMDRV_RTCDRV_NUM_TIMERS ) {
525  return ECODE_EMDRV_RTCDRV_ILLEGAL_TIMER_ID;
526  }
527 
528  INT_Disable();
529  if ( ! timer[ id ].allocated ) {
530  INT_Enable();
531  return ECODE_EMDRV_RTCDRV_TIMER_NOT_ALLOCATED;
532  }
533 
534  if ( timeout == 0 ) {
535  if ( callback != NULL ) {
536  callback( id, user );
537  }
538  INT_Enable();
539  return ECODE_EMDRV_RTCDRV_OK;
540  }
541 
542  cnt = RTC_COUNTERGET();
543 
544  timer[ id ].callback = callback;
545  timer[ id ].ticks = MSEC_TO_TICKS( timeout );
546  if (rtcdrvTimerTypePeriodic == type) {
547  // Calculate compensation value for periodic timers.
548  timer[ id ].periodicCompensationUsec = 1000 * timeout -
549  (timer[ id ].ticks * TICK_TIME_USEC);
550  timer[ id ].periodicDriftUsec = TICK_TIME_USEC/2;
551  }
552  else
553  {
554  // Compensate for the fact that CNT is normally COMP0+1 after a
555  // compare match event on some devices.
556  timer[ id ].ticks -= RTC_ONESHOT_TICK_ADJUST;
557  }
558  // Add one tick in order to compensate if RTC is close to an increment event.
559  timer[ id ].remaining = timer[ id ].ticks + 1;
560  timer[ id ].running = true;
561  timer[ id ].timerType = type;
562  timer[ id ].user = user;
563 
564  if ( inTimerIRQ == true ) {
565  // Exit now, remaining processing will be done in IRQ handler.
566  INT_Enable();
567  return ECODE_EMDRV_RTCDRV_OK;
568  }
569 
570  // StartTimer() may recurse, keep track of recursion level.
571  if ( startTimerNestingLevel < UINT32_MAX ) {
572  startTimerNestingLevel++;
573  }
574 
575  if ( rtcRunning == false ) {
576 
577 #if defined( RTCDRV_USE_RTC )
578  lastStart = ( cnt ) & RTC_COUNTER_MASK;
579 #elif defined( RTCDRV_USE_RTCC )
580  lastStart = cnt;
581 #endif
582 
583  RTC_INTCLEAR( RTC_COMP_INT );
584 
585  compVal = SL_MIN( timer[ id ].remaining, RTC_CLOSE_TO_MAX_VALUE );
586  RTC_COMPARESET( cnt + compVal );
587 
588  // Start the timer system by enabling the compare interrupt.
589  RTC_INTENABLE( RTC_COMP_INT );
590 
591 #if defined( EMODE_DYNAMIC )
592  // When RTC is running, we can not allow EM3 or EM4.
593  if ( sleepBlocked == false ) {
594  sleepBlocked = true;
595  SLEEP_SleepBlockBegin( sleepEM3 );
596  }
597 #endif
598 
599  rtcRunning = true;
600 
601  } else {
602 
603  // The timer system is running. We must stop, update timers with the time
604  // elapsed so far, find the timer with the shortest timeout and then restart.
605  // As StartTimer() may be called from the callbacks we only do this
606  // processing at the first nesting level.
607  if ( startTimerNestingLevel == 1 ) {
608 
609  timer[ id ].running = false;
610  // This loop is repeated if CNT is incremented while processing.
611  do {
612 
613  RTC_INTDISABLE( RTC_COMP_INT );
614 
615  timeElapsed = TIMEDIFF( cnt, lastStart );
616 #if defined( RTCDRV_USE_RTC )
617  // Compensate for the fact that CNT is normally COMP0+1 after a
618  // compare match event.
619  if ( timeElapsed == RTC_MAX_VALUE ) {
620  timeElapsed = 0;
621  }
622 #endif
623 
624  // Update all timers with elapsed time.
625  checkAllTimers( timeElapsed );
626 
627  // Execute timer callbacks.
628  executeTimerCallbacks();
629 
630  // Set timer to running only after checkAllTimers() is called once.
631  if ( loopCnt == 0 ) {
632  timer[ id ].running = true;
633  }
634  loopCnt++;
635 
636  // Restart RTC according to next timeout.
637  rescheduleRtc( cnt );
638 
639  cnt = RTC_COUNTERGET();
640  timeElapsed = TIMEDIFF( cnt, lastStart );
641  timeToNextTimerCompletion = TIMEDIFF( RTC_COMPAREGET(), lastStart );
642 
643  /* If the counter has passed the COMP(ARE) register value since we
644  checked the timers, then we should recheck the timers and reschedule
645  again. */
646  }
647  while ( rtcRunning && (timeElapsed > timeToNextTimerCompletion));
648  }
649  }
650 
651  if ( startTimerNestingLevel > 0 ) {
652  startTimerNestingLevel--;
653  }
654 
655  INT_Enable();
656  return ECODE_EMDRV_RTCDRV_OK;
657 }

705 Ecode_t RTCDRV_TimeRemaining( RTCDRV_TimerID_t id, uint32_t *timeRemaining )
706 {
707  uint64_t tmp;
708  uint32_t timeLeft, currentCnt, lastRtcStart;
709 
710  // Check if valid timer ID.
711  if ( id >= EMDRV_RTCDRV_NUM_TIMERS ) {
712  return ECODE_EMDRV_RTCDRV_ILLEGAL_TIMER_ID;
713  }
714 
715  // Check pointer validity.
716  if ( timeRemaining == NULL ) {
717  return ECODE_EMDRV_RTCDRV_PARAM_ERROR;
718  }
719 
720  INT_Disable();
721  // Check if timer is reserved.
722  if ( ! timer[ id ].allocated ) {
723  INT_Enable();
724  return ECODE_EMDRV_RTCDRV_TIMER_NOT_ALLOCATED;
725  }
726 
727  // Check if timer is running.
728  if ( ! timer[ id ].running ) {
729  INT_Enable();
730  return ECODE_EMDRV_RTCDRV_TIMER_NOT_RUNNING;
731  }
732 
733  timeLeft = timer[ id ].remaining;
734  currentCnt = RTC_COUNTERGET();
735  lastRtcStart = lastStart;
736  INT_Enable();
737 
738  // Get number of RTC clock ticks elapsed since last RTC reschedule.
739  currentCnt = TIMEDIFF( currentCnt, lastRtcStart );
740 
741  if ( currentCnt > timeLeft ) {
742  timeLeft = 0;
743  } else {
744  timeLeft -= currentCnt;
745  }
746 
747  tmp = TICKS_TO_MSEC( timeLeft );
748  *timeRemaining = tmp;
749 
750  return ECODE_EMDRV_RTCDRV_OK;
751 }

902 #if defined( RTCDRV_USE_RTC )
903 void RTC_IRQHandler(void)
904 #elif defined( RTCDRV_USE_RTCC )
905 void RTCC_IRQHandler(void)
906 #endif
907 {
908  uint32_t flags, timeElapsed, cnt, timeToNextTimerCompletion;
909 
910  INT_Disable();
911 
912  // CNT will normally be COMP0+1 at this point,
913  // unless IRQ latency exceeded one tick period.
914 
915  flags = RTC_INTGET();
916 
917  if ( flags & RTC_COMP_INT ) {
918 
919  // Stop timer system by disabling the compare IRQ.
920  // Update all timers with the time elapsed, call callbacks if needed,
921  // then find the timer with the shortest timeout (if any at all) and
922  // reenable the compare IRQ if needed.
923 
924  inTimerIRQ = true;
925 
926  cnt = RTC_COUNTERGET();
927 
928  // This loop is repeated if CNT is incremented while processing.
929  do {
930 
931  RTC_INTDISABLE( RTC_COMP_INT );
932 
933  timeElapsed = TIMEDIFF( cnt, lastStart );
934 
935  // Update all timers with elapsed time.
936  checkAllTimers( timeElapsed );
937 
938  // Execute timer callbacks.
939  executeTimerCallbacks();
940 
941  // Restart RTC according to next timeout.
942  rescheduleRtc( cnt );
943 
944  cnt = RTC_COUNTERGET();
945  timeElapsed = TIMEDIFF( cnt, lastStart );
946  timeToNextTimerCompletion = TIMEDIFF( RTC_COMPAREGET(), lastStart );
947  /* If the counter has passed the COMP(ARE) register value since we
948  checked the timers, then we should recheck the timers and reschedule
949  again. */
950  }
951  while ( rtcRunning && (timeElapsed > timeToNextTimerCompletion));
952  inTimerIRQ = false;
953  }
954 
955 #if defined( EMDRV_RTCDRV_WALLCLOCK_CONFIG )
956  if ( flags & RTC_OF_INT )
957  {
958  RTC_INTCLEAR( RTC_OF_INT );
959  wallClockOverflowCnt++;
960  }
961 #endif
962 
963  INT_Enable();
964 }

966 static void checkAllTimers( uint32_t timeElapsed )
967 {
968  int i;
969 #if defined( EMODE_DYNAMIC )
970  int numOfTimersRunning = 0;
971 #endif
972 
973  // Iterate through the timer table.
974  // Update time remaining, check for timeout and rescheduling of periodic
975  // timers, check for callbacks.
976 
977  for ( i = 0; i < EMDRV_RTCDRV_NUM_TIMERS; i++ ) {
978  timer[ i ].doCallback = false;
979  if ( timer[ i ].running == true ) {
980 #if defined( EMODE_DYNAMIC )
981  numOfTimersRunning++;
982 #endif
983  if ( timer[ i ].remaining > timeElapsed ) {
984  timer[ i ].remaining -= timeElapsed;
985  } else {
986  if ( timer[ i ].timerType == rtcdrvTimerTypeOneshot ) {
987  timer[ i ].running = false;
988 #if defined( EMODE_DYNAMIC )
989  numOfTimersRunning--;
990 #endif
991  } else {
992  // Compensate overdue periodic timers to avoid accumlating errors.
993  timer[ i ].remaining = timer[ i ].ticks - timeElapsed +
994  timer[ i ].remaining;
995  if ( timer[ i ].periodicCompensationUsec > 0 ) {
996  timer[ i ].periodicDriftUsec += timer[i].periodicCompensationUsec;
997  if (timer[ i ].periodicDriftUsec >= TICK_TIME_USEC) {
998  // Add a tick if the timer drift is longer than the time of
999  // one tick.
1000  timer[ i ].remaining += 1;
1001  timer[ i ].periodicDriftUsec -= TICK_TIME_USEC;
1002  }
1003  }
1004  else {
1005  timer[ i ].periodicDriftUsec -= timer[i].periodicCompensationUsec;
1006  if (timer[ i ].periodicDriftUsec >= TICK_TIME_USEC) {
1007  // Subtract one tick if the timer drift is longer than the time
1008  // of one tick.
1009  timer[ i ].remaining -= 1;
1010  timer[ i ].periodicDriftUsec -= TICK_TIME_USEC;
1011  }
1012  }
1013  }
1014  if ( timer[ i ].callback != NULL ) {
1015  timer[ i ].doCallback = true;
1016  }
1017  }
1018  }
1019  }
1020 
1021 #if defined( EMODE_DYNAMIC )
1022  // If no timers are running, we can remove block on EM3 and EM4 sleep modes.
1023  if ( ( numOfTimersRunning == 0 ) && ( sleepBlocked == true ) ) {
1024  sleepBlocked = false;
1025  SLEEP_SleepBlockEnd( sleepEM3 );
1026  }
1027 #endif
1028 }

1031 {

1033  volatile uint32_t now;
1034

1038  now = RTC_COUNTERGET();
1039  } while ( TIMEDIFF( now, startTime ) < ticks );
1040  }

1054 static void rescheduleRtc( uint32_t rtcCnt )
1055 {
1056  int i;
1057  uint64_t min = UINT64_MAX;
1058 
1059  // Find the timer with shortest timeout.
1060  for ( i = 0; i < EMDRV_RTCDRV_NUM_TIMERS; i++ ) {
1061  if ( ( timer[ i ].running == true )
1062  && ( timer[ i ].remaining < min ) ) {
1063  min = timer[ i ].remaining;
1064  }
1065  }
1066 
1067  rtcRunning = false;
1068  if ( min != UINT64_MAX ) {
1069  min = SL_MIN( min, RTC_CLOSE_TO_MAX_VALUE );
1070 #if defined( RTCDRV_USE_RTC )
1071  if ( inTimerIRQ == false ) {
1072  lastStart = ( rtcCnt ) & RTC_COUNTER_MASK;
1073  } else
1074 #endif
1075  {
1076  lastStart = rtcCnt;
1077  }
1078  RTC_INTCLEAR( RTC_COMP_INT );
1079 
1080  RTC_COMPARESET( rtcCnt + min );
1081 
1082 #if defined( EMODE_DYNAMIC )
1083  // When RTC is running, we can not allow EM3 or EM4.
1084  if ( sleepBlocked == false ) {
1085  sleepBlocked = true;
1086  SLEEP_SleepBlockBegin( sleepEM3 );
1087  }
1088 #endif
1089 
1090  rtcRunning = true;
1091 
1092  // Reenable compare IRQ.
1093  RTC_INTENABLE( RTC_COMP_INT );
1094  }
1095 }
