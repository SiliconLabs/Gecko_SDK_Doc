218 void TIMER_Reset(TIMER_TypeDef *timer)
219 {
220  int i;
221 
222  EFM_ASSERT(TIMER_REF_VALID(timer));
223 
224  /* Make sure disabled first, before resetting other registers */
225  timer->CMD = TIMER_CMD_STOP;
226 
227  timer->CTRL = _TIMER_CTRL_RESETVALUE;
228  timer->IEN = _TIMER_IEN_RESETVALUE;
229  timer->IFC = _TIMER_IFC_MASK;
230  timer->TOP = _TIMER_TOP_RESETVALUE;
231  timer->TOPB = _TIMER_TOPB_RESETVALUE;
232  timer->CNT = _TIMER_CNT_RESETVALUE;
233  /* Do not reset route register, setting should be done independently */
234  /* (Note: ROUTE register may be locked by DTLOCK register.) */
235 
236  for (i = 0; TIMER_CH_VALID(i); i++)
237  {
238  timer->CC[i].CTRL = _TIMER_CC_CTRL_RESETVALUE;
239  timer->CC[i].CCV = _TIMER_CC_CCV_RESETVALUE;
240  timer->CC[i].CCVB = _TIMER_CC_CCVB_RESETVALUE;
241  }
242 
243  /* Reset dead time insertion module, no effect on timers without DTI */
244 
245 #if defined(TIMER_DTLOCK_LOCKKEY_UNLOCK)
246  /* Unlock DTI registers first in case locked */
247  timer->DTLOCK = TIMER_DTLOCK_LOCKKEY_UNLOCK;
248 
249  timer->DTCTRL = _TIMER_DTCTRL_RESETVALUE;
250  timer->DTTIME = _TIMER_DTTIME_RESETVALUE;
251  timer->DTFC = _TIMER_DTFC_RESETVALUE;
252  timer->DTOGEN = _TIMER_DTOGEN_RESETVALUE;
253  timer->DTFAULTC = _TIMER_DTFAULTC_MASK;
254 #endif
255 }
